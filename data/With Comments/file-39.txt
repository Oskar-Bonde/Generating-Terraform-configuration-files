# Terraform Block with AWS
terraform {
  required_version = "~> 0.14"
  required_providers {
    aws = {
      source  = "hashicorp/aws"
      version = "~> 3.0"
    }
  }
}

# Provider Block AWS with region var.aws_region and default profile
provider "aws" {
  region  = var.aws_region
  profile = "default"
}

# Create S3 Bucket Resource called s3_bucket. 
resource "aws_s3_bucket" "s3_bucket" {
  bucket = var.bucket_name
  acl    = "public-read"
  policy = <<EOF
  {
  "Version": "2012-10-17",
  "Statement": [
      {
          "Sid": "PublicReadGetObject",
          "Effect": "Allow",
          "Principal": "*",
          "Action": [
              "s3:GetObject"
          ],
          "Resource": [
              "arn:aws:s3:::${var.bucket_name}/*"
          ]
      }
    ]
  }
EOF
  website {
    index_document = "index.html"
    error_document = "error.html"
  }
  tags          = var.tags
  force_destroy = true
}


# Output block: ARN of the S3 Bucket
output "arn" {
  value       = aws_s3_bucket.s3_bucket.arn
}

# Output block: Name (id) of the bucket
output "name" {
  value       = aws_s3_bucket.s3_bucket.id
}

# Output block: Domain Name of the bucket
output "domain" {
  value       = aws_s3_bucket.s3_bucket.website_domain
}

# Output block: Endpoint Information of the bucket
output "endpoint" {
  value       = aws_s3_bucket.s3_bucket.website_endpoint
}
