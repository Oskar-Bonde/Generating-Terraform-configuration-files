# raw-data/hashicorp-certified-terraform\02-Terraform-Basics\02-02-Terraform-Command-Basics\terraform-manifests
# Terraform Block with the AWS provider
terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
      #version = "~> 3.21" # Optional but recommended in production
    }
  }
}

# Provider block with AWS in us-east-1 region
provider "aws" {
  profile = "default" # AWS Credentials Profile configured on your local desktop terminal  $HOME/.aws/credentials
  region  = "us-east-1"
}

# Resource block with an EC2 instance called ec2demo with the t2.micro instance type and ami-0ff8a91507f77f867
resource "aws_instance" "ec2demo" {
  ami           = "ami-0ff8a91507f77f867" 
  instance_type = "t2.micro"
}
