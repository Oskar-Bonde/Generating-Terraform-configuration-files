terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider Block: aws in us east 1 
provider "aws" {
  region = "us-east-1"
}

# Create a AWS VPC resource with private network 10.0.0.0/16
resource "aws_vpc" "my_vpc" {
  cidr_block = "10.0.0.0/16"
}

# create a default network acl, vpc acl id. As ingress, set protocal -1, rule number 100, cird block 0.0.0.0/0, from port 83 to port 0 and action allow
resource "aws_default_network_acl" "my_acl" {
  default_network_acl_id = aws_vpc.my_vpc.default_network_acl_id
  ingress {
    protocol = -1
    rule_number = 100
    cidr_block = "0.0.0.0/0"
    from_port = 83
    to_port = 0
    action = "allow"
  }
}
