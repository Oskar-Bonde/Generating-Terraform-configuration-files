provider
terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider AWS region var.aws_region
variable "aws_region" {
  default = "us-east-1"
}

# Create Variable that is a map of strings. The default is that key Terraform maps to true and Environment key maps to dev
variable "aws_credentials" {
  type = "map"

  default = {
    aws_access_key = ""
    aws_secret_key = ""
  }
}
