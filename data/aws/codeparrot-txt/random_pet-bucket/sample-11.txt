provider "aws" {
  profile = "default"
  region  = "us-east-1"
}

# AWS provider block in region us-east-1
resource "aws_instance" "aws" {
  ami           = "ami-0ff8a91507f77f867" 
  instance_type = "t2.micro"
}

# Resource block to create a random pet name of length 5 with separator - [A-Za-z0-9]
resource "random_pet_name" "terraform-example-5" {
  keepers = {
    keeper_name = "terraform-example-5-${random_pet_name.terraform-example-5.id}"
  }
  byte_length = 5
}

# Resource Block: Create AWS S3 Bucket with bucket set as random pet name
resource "aws_s3_bucket" "terraform-example-bucket" {
  bucket = "terraform-example-bucket"
  acl    = "private"
  force_destroy = true
}
