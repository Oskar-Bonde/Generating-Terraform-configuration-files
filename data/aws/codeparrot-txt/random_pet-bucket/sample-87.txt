provider "aws" {
  profile = "default"
  region  = "us-east-1"
}

# AWS provider block in region us-east-1
provider "aws" {
  alias  = "us-east-1"
  region = "us-east-1"
}

# Resource block to create a random pet name of length 5 with separator -
# this is to ensure unique resource names
resource "random_pet_name" "terraform-example-instance" {
  byte_length = 5
}

# Resource Block: Create AWS S3 Bucket with bucket set as random pet name
resource "aws_s3_bucket" "terraform-example-instance" {
  bucket = "terraform-example-instance"
  acl    = "private"

  tags {
    Name = "terraform-example-instance"
  }
}
