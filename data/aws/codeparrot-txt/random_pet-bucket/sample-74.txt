provider "aws" {
  profile = "default"
  region  = "us-east-1"
}

# AWS provider block in region us-east-1
resource "aws_instance" "aws" {
  ami           = "ami-0ff8a91507f77f867" 
  instance_type = "t2.micro"
}

# Resource block to create a random pet name of length 5 with separator - [project_id]-[environment]-[region]
resource "random_pet_name" "suffix" {
  length  = 5
  special = false
  upper   = false
}

# Resource Block: Create AWS S3 Bucket with bucket set as random pet name
resource "aws_s3_bucket" "bucket" {
  bucket = "random-pet-bucket-${random_pet_name.suffix.result}"
  acl    = "private"

  tags {
    Name = "random-pet-bucket-${random_pet_name.suffix.result}"
  }
}
