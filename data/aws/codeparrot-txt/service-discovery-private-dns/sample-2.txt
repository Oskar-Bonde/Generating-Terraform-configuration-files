terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider Block: aws in us east 1 region and default profile
provider "aws" {
  profile = "default"
  region  = "us-east-1"
}

# Create a VPC resource with cidr block 10.0.0.0/16. Enable DNS support and DNS hostnames
resource "aws_vpc" "default" {
  cidr_block           = "10.0.0.0/16"
  enable_dns_hostnames = true
  tags = {
    Name = "terraform-example-vpc"
  }
}

# Make a service discovery private dns namespace resource. Give it the name 'example.terraform.local' and use vpc id as the value
resource "aws_route_table" "private-us-east-1a-private" {
  vpc_id = aws_vpc.default.id
  tags = {
    Name = "terraform-example-vpc-private-us-east-1a.private"
  }
}
