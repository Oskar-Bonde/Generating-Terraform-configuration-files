provider
terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider Block with region us-east-1
provider "aws" {
  region = "us-east-1"
}

# Create S3 Bucket Resource. Set bucket to cookievariable bucket_prefix
resource "aws_s3_bucket" "bucket_prefix" {
  bucket = "cookievariable-${var.env}"
  acl    = "public-read"
}

# Resource, aws s3 bucket object. Set key to index.html and use the s3 bucket id
resource "aws_s3_bucket" "index_html" {
  bucket = "index.html"
  acl    = "public-read"
}
