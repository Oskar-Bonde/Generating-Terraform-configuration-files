terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider aws block with us east 1 region
provider "aws" {
  region = "us-east-1"
}

# Create S3 Bucket, set bucket to  "icecream"
# and upload it to S3 bucket
resource "aws_s3_bucket" "s3_bucket" {
  bucket = "icecream-terraform-state"
  acl    = "private"
  force_destroy = true
}
