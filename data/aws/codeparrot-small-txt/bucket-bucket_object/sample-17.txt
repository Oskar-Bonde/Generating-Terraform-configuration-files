provider
terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider Block with region us-east-1
provider "aws" {
  region = "us-east-1"
}

# Create S3 Bucket Resource. Set bucket to cookie file name
resource "aws_s3_bucket" "terraform_state" {
  bucket = "terraform-state"
  acl    = "private"
}
