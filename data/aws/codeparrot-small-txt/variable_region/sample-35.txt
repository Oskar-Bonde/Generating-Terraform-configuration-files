terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider AWS in region var.aws_region
provider "aws" {
  region = var.aws_region
}

# Create variable called aws_region that is a string typpe with default value "us-east-1"
variable "aws_region" {
  description = "AWS region"
  default = "us-east-1"
}
