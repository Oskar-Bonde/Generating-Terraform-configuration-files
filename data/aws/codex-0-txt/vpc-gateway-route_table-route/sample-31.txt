terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider Block: aws in us east 1
provider "aws" {
  region = "us-east-1"
}

# Create a AWS VPC resource with private network 10.0.0.0/16
resource "aws_vpc" "vpc_example" {
  cidr_block = "10.0.0.0/16"
}

# AWS Internet Gateway with vpc id 
resource "aws_internet_gateway" "igw_example" {
  vpc_id = aws_vpc.vpc_example.id
}

# Create AWS route table with vpc id
resource "aws_route_table" "rt_example" {
  vpc_id = aws_vpc.vpc_example.id
}

# Create route resource. Use destination 0.0.0.0/0, gateway id and route talbe id
resource "aws_route" "route_example" {
  route_table_id = aws_route_table.rt_example.id
  destination_cidr_block = "0.0.0.0/0"
  gateway_id = aws_internet_gateway.igw_example.id
}
