  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }

}

# Provider AWS in region "us-east-1"
provider "aws" {
  profile = "default"
  region  = "us-east-1"
}

# Create Security Group that allows port 80, 443 inbound and all ports outbound

  name        = "default"
  description = "Allow all outbound traffic"
