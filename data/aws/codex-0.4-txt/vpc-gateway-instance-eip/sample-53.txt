terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider Block: aws in us east 1
provider "aws" {
  region = "us-east-1"
}

# Create a AWS VPC resource with private network 10.0.0.0/16
resource "aws_vpc" "private_vpc" {
  cidr_block = "10.0.0.0/16"
  enable_dns_hostnames = true
  enable_dns_support = true
  tags = {
    Name = "private_vpc"
  }
}

# AWS Internet Gateway with vpc id
resource "aws_internet_gateway" "private_vpc_igw" {
  vpc_id = aws_vpc.private_vpc.id
  tags = {
    Name = "private_vpc_igw"
  }
}

# Create EC2 Instance, use ami-047a51fa27710816e and t2.micro
resource "aws_instance" "private_vpc_ec2" {
  ami = "ami-047a51fa27710816e"
  instance_type = "t2.micro"
  vpc_security_group_ids = [aws_security_group.private_vpc_sg.id]
  subnet_id = aws_subnet.private_vpc_subnet.id
  tags = {
    Name = "private_vpc_ec2"
  }
}

# Resource block: Create Elastic IP that uses vpc, ec2 instance id and depends on internet gateway
resource "aws_eip" "private_vpc_eip" {
  vpc = true
  instance = aws_instance.private_vpc_ec2.id
  depends_on = [aws_internet_gateway.private_vpc_igw]
}
