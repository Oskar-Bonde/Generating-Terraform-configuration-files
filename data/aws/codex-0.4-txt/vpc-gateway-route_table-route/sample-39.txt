terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider Block: aws in us east 1
provider "aws" {
  region = "us-east-1"
}

# Create a AWS VPC resource with private network 10.0.0.0/16
resource "aws_vpc" "private_vpc" {
  cidr_block = "10.0.0.0/16"
  enable_dns_hostnames = true
  enable_dns_support = true
  tags = {
    Name = "private_vpc"
  }
}

# AWS Internet Gateway with vpc id 
resource "aws_internet_gateway" "private_igw" {
  vpc_id = aws_vpc.private_vpc.id
  tags = {
    Name = "private_igw"
  }
}

# Create AWS route table with vpc id
resource "aws_route_table" "private_rt" {
  vpc_id = aws_vpc.private_vpc.id
  tags = {
    Name = "private_rt"
  }
}

# Create route resource. Use destination 0.0.0.0/0, gateway id and route talbe id
resource "aws_route" "private_route" {
  route_table_id = aws_route_table.private_rt.id
  destination_cidr_block = "0.0.0.0/0"
  gateway_id = aws_internet_gateway.private_igw.id
}
