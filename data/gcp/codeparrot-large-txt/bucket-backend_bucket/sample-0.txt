terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block with only region set to europe north1
provider "google" {
}

# create storage bucket resource, give it a name and set location to europe-north1
resource "google_storage_bucket" "bucket" {
  name = "my-bucket"
  location = "europe-north1"
}

# create compute backend bucket, give it a name, use bucket name from storage bucket and enable cdn-account-key-creation
resource "google_storage_bucket" "bucket_with_cdn" {
  name = "my-bucket-with-cdn"
  location = "europe-north1"
  force_destroy = true
}
