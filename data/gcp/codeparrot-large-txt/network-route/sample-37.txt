terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block with only region set to europe north1
provider "google" {
}

# create compute network, give it a name and auto create subnetworks
resource "google_compute_network" "default" {
  name = "my-network"
}

# compute route with dest range 15.0.0.0/24 and next hop id 10.132.1.5. Use the compute network id and set priority to 100 to get a better name
resource "google_compute_route" "default" {
  name       = "default-route"
  network    = google_compute_network.default.id
  dest_range = "10.132.1.5/24"
  priority   = 100

  next_hop_vpn_tunnel = google_compute_vpn_tunnel.default.id
}
