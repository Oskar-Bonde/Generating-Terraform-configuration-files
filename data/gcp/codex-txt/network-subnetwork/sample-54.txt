terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block with only region set to europe north1
provider "google" {
  region = "europe-north1"
}

# create compute network with name surf
resource "google_compute_network" "surf" {
  name                    = "surf"
  auto_create_subnetworks = "true"
}

# create compute subnetwork. Use ip cidr range 10.0.0.0/16 and use compute network surf id
resource "google_compute_subnetwork" "surf" {
  name          = "surf"
  ip_cidr_range = "10.0.0.0/16"
  network       = google_compute_network.surf.id
}
