}

  name = "tf-test-%s"
  description = "Test notebook instance"
  machine_type = "n1-standard-2"
  vm_image = "projects/deeplearning-platform-release/global/images/family/tf-latest-cpu"
  region = "us-west1"
  boot_disk_size_gb = 10
  labels = {
    env = "test"
  }
