terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block with only region set to europe north1
provider "google" {
}

# create storage bucket resource, give it a name and set location to europe-north1
resource "google_storage_bucket" "bucket" {
  name = "my-bucket"
  location = "europe-north1"
}

# create compute backend bucket, give it a name, use bucket name from storage bucket and enable cdnvariable
resource "google_storage_bucket" "bucket_cdn" {
  name = "my-bucket-cdn"
  location = "europe-north1"
  cdn_configuration {
    cdn_token {
      enabled = true
      value = "https://cdn-a.storage.googleapis.com/${google_storage_bucket.bucket.name}"
    }
  }
}
