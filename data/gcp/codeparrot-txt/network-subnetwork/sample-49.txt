terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block with only region set to europe north1
provider "google" {
}

# create compute network with name surfly-1
resource "google_compute_network" "surfly-1" {
  name = "surfly-1"
}

# create compute subnetwork. Use ip cidr range 10.0.0.0/16 and use compute network surf id
resource "google_compute_subnetwork" "surfly-1" {
  name = "surfly-1"
  ip_cidr_range = "10.0.0.0/16"
  network = google_compute_network.surfly-1.self_link
}
