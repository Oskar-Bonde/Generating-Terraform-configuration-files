terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block with only region set to europe north1
provider "google" {
}

# create compute network with a name surfly-name
resource "google_compute_network" "surfly-name" {
  name = "surfly-name"
}

# create compute subnetwork. Use ip cidr range 10.0.0.0/16 and use previous compute network called surf
resource "google_compute_subnetwork" "surfly-subnet" {
  name = "surfly-subnet"
  ip_cidr_range = "10.0.0.0/16"
  network = google_compute_network.surfly-name.self_link
}

# resource compute address, set subnetwork id. Set address type to internal, use address 10.0.42.42
resource "google_compute_address" "surfly-address" {
  name = "surfly-address"
}
