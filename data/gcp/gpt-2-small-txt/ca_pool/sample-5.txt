resource "google_ca_certificate" "ca" {
  provider = google-beta
  name    = "my-ca-certificate"
  location = "europe-west4"

  validation {
    domains = ["my-ca.com"]
  }

  lifecycle {
    ignore_changes = ["name"]
  }
}
