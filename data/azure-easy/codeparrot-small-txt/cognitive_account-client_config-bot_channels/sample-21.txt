terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azurerm
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg2" {
  name     = "myTFResourceGroup2"
  location = "westus2"
}

# create cognitive account resource
resource "azurerm_resource_group" "azure_sa" {
  name     = "myTFResourceGroup3"
  location = "westus2"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# create storage account
resource "azurerm_storage_account" "azure_sa" {
  name                     = "mytfstorageaccount"
  resource_group_name      = azurerm_resource_group.rg.name
  location                 = "westus2"
  account_tier             = "Standard"
  account_replication_type = "LRS"
}

# create bot channels registration resourceresource
resource "azurerm_resource_group" "azure_sa_channel" {
  name     = "mytfchannelregistrationresource"
  location = "westus2"
}
