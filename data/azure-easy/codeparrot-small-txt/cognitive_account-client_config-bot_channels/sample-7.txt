terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azurerm
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg_test" {
  name     = "myTFResourceGroup"
  location = "westus2"
}

# create cognitive account resourcevariable
resource "azurerm_resource_group" "azure_cognitive_account_resource_group" {
  name     = "azure_cognitive_account_resource_group"
  location = "westus2"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# create storage account resource variable
resource "azurerm_storage_account" "azure_account_data" {
  name                     = "mytfstorageaccount"
  resource_group_name      = azurerm_resource_group.rg_test.name
  location                 = "westus2"
  account_tier             = "Standard"
  account_replication_type = "GRS"
}

# create bot channels registration resourcevariable
resource "azurerm_resource_group" "azure_cognitive_channel_registration_resource_group" {
  name     = "azure_cognitive_channel_registration_resource_group"
