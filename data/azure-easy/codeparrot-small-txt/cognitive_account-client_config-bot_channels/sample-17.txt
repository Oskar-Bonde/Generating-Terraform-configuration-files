terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azurerm
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg_dev" {
  name     = "myTFResourceGroup_dev"
  location = "westus2"
}

# create cognitive account resource
resource "azurerm_resource_group" "rg_dev_dev" {
  name     = "myTFResourceGroup_dev_dev"
  location = "westus2"
}

# data block, azurem client configvariable
data "azurerm_client_configvariable" "current" {}

# create storage account resource
resource "azurerm_storage_account" "stor" {
  name                     = "mytfstorageaccount"
  resource_group_name      = azurerm_resource_group.rg_dev.name
  location                 = "westus2"
  account_tier             = "Standard"
  account_replication_type = "GRS"

  tags = {
    environment = "dev"
  }
}

# create bot channels registration resource
resource "azurerm_resource_group" "rg_dev_dev_dev" {
  name     = "myTFResourceGroup_dev_dev"
  location = "westus2"
}
