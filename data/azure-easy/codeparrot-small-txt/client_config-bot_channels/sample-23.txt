terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azurerm
provider "azurerm" {
  features {}
}

# create resource groupresource
resource "azurerm_resource_group_resource" "rg" {
  name     = "myTFResourceGroupResource"
  location = "westus2"
  resource_group_name = azurerm_resource_group.rg.name
  tags = {
    env = "dev"
  }
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# create storage account
resource "azurerm_storage_account" "stor" {
  name                     = "mytfstorageaccount"
  resource_group_name      = azurerm_resource_group.rg.name
  location                 = "westus2"
  account_tier             = "Standard"
  account_replication_type = "GRS"
}

# create bot channels registration resource
resource "azurerm_registration_service" "reg" {
  name                = "mytfregistrationservice"
  location            = "westus2"
  resource_group_name = azurerm_resource_group.rg.name
  sku                 = "Standard"
  account_tier        = "Standard"
  account_replication_type = "GRS"
  webhook_url          = "https://github.com/webhook-service-account/webhook"

  # subscription_id = "REPLACE-WITH-YOUR-SUBSCRIPTION-ID"
}
