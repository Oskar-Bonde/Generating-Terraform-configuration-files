terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azurerm
provider "azurerm" {
  features {}
}

# create resource groupvariable
resource "azurerm_resource_group_variable" "rg_var" {
  name     = "myTFResourceGroup"
  location = "westus2"
  value     = "test"
}

# create cognitive account resource
resource "azurerm_resource_group_role" "rg_role" {
  name                = "azurerm_rg_role"
  location            = "westus2"
  resource_group_name = azurerm_resource_group.rg.name
  role_type            = "Standard_LRS"
  account_tier             = "Standard"
}

# data block, azurem client configresource
data "azurerm_client_config" "current" {}

# create storage account resource
resource "azurerm_storage_account" "stor" {
  name                = "mytfstorageaccount"
  resource_group_name = azurerm_resource_group.rg.name
  location            = "westus2"
  account_tier             = "Standard"
  account_replication_type = "LRS"
}

# create bot channels registration resource
resource "azurerm_resource_group_membership" "rg_member" {
  resource_group_name = azurerm_resource_group.rg.name
  location            = "westus2"
  membership_type     = "Subscription"
  members             = [azurerm_resource_group.rg_member.name]
}

# resource block called bot channel direct line speechresource
resource "azurerm_resource_group" "rg" {
  name     = "myTFResourceGroup"
  location = "westus2"
}
