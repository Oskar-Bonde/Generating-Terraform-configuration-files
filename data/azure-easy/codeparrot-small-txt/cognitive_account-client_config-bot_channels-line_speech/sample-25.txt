terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azurerm
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg_test" {
  name     = "myTFResourceGroup"
  location = "westus2"
}

# create cognitive account resourcevariable
resource "azurerm_resource_group" "azure_cognitive_account" {
  name     = "myTFResourceGroup"
  location = "westus2"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# create storage account
resource "azurerm_storage_account" "azure_account" {
  name                     = "mytfstorageaccount"
  resource_group_name      = azurerm_resource_group.rg_test.name
  location                 = "westus2"
  account_tier             = "Standard"
  account_replication_type = "LRS"
}

# create bot channels registration resourcevariable
resource "azurerm_resource_group" "azure_cognitive_channel" {
  name     = "myTFResourceGroup"
  location = "westus2"
}

# resource block called bot channel direct line speech
resource "azurerm_resource_group" "azure_cognitive_channel_speech" {
  name     = "myTFResourceGroup"
  location = "westus2"
}
