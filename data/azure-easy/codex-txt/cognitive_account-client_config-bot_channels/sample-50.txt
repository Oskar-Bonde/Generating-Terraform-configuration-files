terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "rg-name"
  location = "westus"
}

# create cognitive account resource
resource "azurerm_cognitive_account" "cog" {
  name                = "cog-name"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku_name            = "S0"
  kind                = "TextAnalytics"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# data block, azurem cognitive account keys
data "azurerm_cognitive_account_keys" "cog" {
  name                = azurerm_cognitive_account.cog.name
  resource_group_name = azurerm_resource_group.rg.name
}

# create bot channels registration resource
resource "azurerm_bot_channels_registration" "bot" {
  name                = "bot-name"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku                 = "F0"
  kind                = "sdk"
  microsoft_app_id    = "app-id"
  microsoft_app_password = "app-password"
  endpoint            = "https://bot-name.azurewebsites.net/api/messages"
  developer_app_insights_key = "app-insights-key"
  developer_app_insights_api_key = "app-insights-api-key"
  developer_app_insights_application_id = "app-insights-application-id"
  cognitive_models {
    qna {
      kb_id = "kb-id"
      subscription_key = data.azurerm_cognitive_account_keys.cog.primary_access_key
      endpoint_host_name = data.azurerm_cognitive_account.cog.endpoint
    }
  }
}
