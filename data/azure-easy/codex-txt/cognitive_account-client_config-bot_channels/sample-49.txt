terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "rg-terraform-demo"
  location = "eastus"
}

# create cognitive account resource
resource "azurerm_cognitive_account" "cog_acc" {
  name                = "cog-acc-terraform-demo"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  kind                = "TextAnalytics"
  sku_name            = "S0"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# create cognitive key resource
resource "azurerm_cognitive_account_key" "cog_acc_key" {
  name                = azurerm_cognitive_account.cog_acc.name
  resource_group_name = azurerm_resource_group.rg.name
  key_name            = "key1"
  key_vault_uri       = "https://tf-demo-kv.vault.azure.net/"
  key_vault_client_id = data.azurerm_client_config.current.client_id
  key_vault_client_secret = data.azurerm_client_config.current.client_secret
}

# create bot channels registration resource
resource "azurerm_bot_channels_registration" "bot_reg" {
  name                = "bot-reg-terraform-demo"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku                 = "F0"
  kind                = "sdk"
  microsoft_app_id    = "app-id-terraform-demo"
  microsoft_app_password = "app-password-terraform-demo"
}
