terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "rg-terraform-demo"
  location = "West Europe"
}

# create cognitive account resource
resource "azurerm_cognitive_account" "cognitive" {
  name                = "cognitive-terraform-demo"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku_name            = "S0"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# create cognitive account key
resource "azurerm_cognitive_account_key" "cognitive" {
  name                = azurerm_cognitive_account.cognitive.name
  resource_group_name = azurerm_resource_group.rg.name
  key_name            = "key1"
}

# create bot channels registration resource
resource "azurerm_bot_channels_registration" "bot" {
  name                = "bot-terraform-demo"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku                 = "F0"
  kind                = "bot"
  microsoft_app_id    = data.azurerm_client_config.current.object_id
  endpoint            = "https://bot-terraform-demo.azurewebsites.net/api/messages"
  msi_endpoint        = "https://bot-terraform-demo.azurewebsites.net/api/messages"
  msi_secret          = "secret"
}
