terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "myResourceGroup"
  location = "West US"
}

# create cognitive account resource
resource "azurerm_cognitive_account" "cog" {
  name                = "myCognitiveAccount"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  kind                = "TextAnalytics"
  sku_name            = "S0"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# data block, azurem cognitive account keys
data "azurerm_cognitive_account_keys" "cog" {
  name                = azurerm_cognitive_account.cog.name
  resource_group_name = azurerm_resource_group.rg.name
}

# create bot channels registration resource
resource "azurerm_bot_channels_registration" "bot" {
  name                = "myBot"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku                 = "F0"
  microsoft_app_id    = "00000000-0000-0000-0000-000000000000"
  microsoft_app_password = "password"
  endpoint            = "https://mybot.azurewebsites.net/api/messages"
  developer_app_insights_key = "00000000-0000-0000-0000-000000000000"
  developer_app_insights_api_key = "00000000-0000-0000-0000-000000000000"
  developer_app_insights_application_id = "00000000-0000-0000-0000-000000000000"
  cognitive_models {
    dispatch_model_id = "00000000-0000-0000-0000-000000000000"
    luis_app_id       = "00000000-0000-0000-0000-000000000000"
    luis_authoring_key = data.azurerm_cognitive_account_keys.cog.key1
    luis_authoring_endpoint = data.azurerm_client_config.current.endpoints.luis_authoring_endpoint
    luis_endpoint_key = data.azurerm_cognitive_account_keys.cog.key1
    luis_endpoint = data.azurerm_client_config.current.endpoints.luis_endpoint
  }
}
