terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "rg-name"
  location = "westus"
}

# create cognitive account resource
resource "azurerm_cognitive_account" "cognitive_account" {
  name                = "cognitive-account-name"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  kind                = "TextAnalytics"
  sku_name            = "S0"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# data block, azurem cognitive account keys
data "azurerm_cognitive_account_keys" "cognitive_account_keys" {
  name                = azurerm_cognitive_account.cognitive_account.name
  resource_group_name = azurerm_resource_group.rg.name
}

# create bot channels registration resource
resource "azurerm_bot_channels_registration" "bot_channels_registration" {
  name                = "bot-channels-registration-name"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku                 = "F0"
  kind                = "sdk"
  microsoft_app_id    = "microsoft-app-id"
  microsoft_app_password = "microsoft-app-password"
  endpoint            = "https://endpoint.com"
  developer_app_insights_key = "developer-app-insights-key"
  developer_app_insights_api_key = "developer-app-insights-api-key"
  developer_app_insights_application_id = "developer-app-insights-application-id"
  cognitive_models {
    cognitive_model_id = "cognitive-model-id"
    version = "cognitive-model-version"
    cognitive_model_endpoint_key = data.azurerm_cognitive_account_keys.cognitive_account_keys.key1
    cognitive_model_endpoint = azurerm_cognitive_account.cognitive_account.endpoint
  }
  tags = {
    environment = "Production"
  }
}
