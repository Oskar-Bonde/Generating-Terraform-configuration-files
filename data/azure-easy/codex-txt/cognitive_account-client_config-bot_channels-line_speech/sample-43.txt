terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "myResourceGroup"
  location = "westus"
}

# create cognitive account resource
resource "azurerm_cognitive_account" "cog" {
  name                = "myCognitiveAccount"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  kind                = "TextAnalytics"
  sku_name            = "S0"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# data block, azurem cognitive account key
data "azurerm_cognitive_account_key" "cog" {
  name                = azurerm_cognitive_account.cog.name
  resource_group_name = azurerm_resource_group.rg.name
}

# create bot channels registration resource
resource "azurerm_bot_channels_registration" "bot" {
  name                = "myBotChannelsRegistration"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku                 = "F0"
  microsoft_app_id    = "00000000-0000-0000-0000-000000000000"
  microsoft_app_password = "password"
  endpoint            = "https://myBotChannelsRegistration.azurewebsites.net/api/messages"
  developer_app_insights_key = "00000000-0000-0000-0000-000000000000"
  developer_app_insights_api_key = "00000000-0000-0000-0000-000000000000"
  cognitive_models {
    qna {
      name = "QnA"
      subscription_key = data.azurerm_cognitive_account_key.cog.key1
      kb_id = "00000000-0000-0000-0000-000000000000"
    }
  }
}

# resource block called bot channel direct line speech
resource "azurerm_bot_channel_direct_line_speech" "bot" {
  name                = "myBotChannelDirectLineSpeech"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  bot_name            = azurerm_bot_channels_registration.bot.name
  bot_resource_group  = azurerm_bot_channels_registration.bot.resource_group_name
  subscription_key    = data.azurerm_client_config.current.subscription_id
  speech_subscription_key = data.azurerm_cognitive_account_key.cog.key1
  speech_region       = azurerm_resource_group.rg.location
}
