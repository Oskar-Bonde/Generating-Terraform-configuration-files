terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "my-rg"
  location = "westus"
}

# create cognitive account resource
resource "azurerm_cognitive_account" "cog" {
  name                = "my-cog-account"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku_name            = "S0"
  kind                = "TextAnalytics"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# data block, cognitive account key
data "azurerm_cognitive_account_key" "cog" {
  name                = azurerm_cognitive_account.cog.name
  resource_group_name = azurerm_resource_group.rg.name
}

# create bot channels registration resource
resource "azurerm_bot_channels_registration" "bot" {
  name                = "my-bot-channels-registration"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku                 = "F0"
  kind                = "sdk"
  microsoft_app_id    = "my-app-id"
  microsoft_app_password = "my-app-password"
  endpoint            = "https://my-bot-channels-registration.azurewebsites.net/api/messages"
  developer_app_insights_key = "my-app-insights-key"
  developer_app_insights_api_key = "my-app-insights-api-key"
  cognitive_models {
    dispatch_model_id = "my-dispatch-model-id"
    version = "0.1"
    cognitive_model {
      model_id = "my-model-id"
      model_version = "0.1"
      cognitive_services {
        account_id = azurerm_cognitive_account.cog.id
        subscription_key = data.azurerm_cognitive_account_key.cog.key1
        region = azurerm_cognitive_account.cog.location
      }
    }
  }
}

# resource block called bot channel direct line speech
resource "azurerm_bot_channel_directline_speech" "bot" {
  name                = "my-bot-channel-directline-speech"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  bot_channels_registration_id = azurerm_bot_channels_registration.bot.id
  speech_subscription_key = data.azurerm_client_config.current.client_secret
  speech_region = azurerm_bot_channels_registration.bot.location
}
