terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "my-rg"
  location = "westus"
}

# create cognitive account resource
resource "azurerm_cognitive_account" "cog" {
  name                = "my-cog-account"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku_name            = "S0"
  kind                = "CognitiveServices"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# create a service principal
resource "azurerm_azuread_application" "app" {
  name = "my-app"
}

# create bot channels registration resource
resource "azurerm_bot_channels_registration" "bot" {
  name                = "my-bot"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku                 = "F0"
  kind                = "bot"
  microsoft_app_id    = azurerm_azuread_application.app.application_id
  endpoint            = "https://my-bot.azurewebsites.net/api/messages"
  developer_app_insights_api_key = azurerm_cognitive_account.cog.primary_access_key
  developer_app_insights_application_id = azurerm_cognitive_account.cog.primary_access_key
  tags = {
    environment = "Production"
  }
}

# resource block called bot channel direct line speech
resource "azurerm_bot_channel_direct_line_speech" "bot" {
  name                = "my-bot-channel"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  bot_id              = azurerm_bot_channels_registration.bot.id
  cognitive_services_authoring_key = azurerm_cognitive_account.cog.primary_access_key
  cognitive_services_authoring_region = azurerm_cognitive_account.cog.location
  client_secret       = data.azurerm_client_config.current.client_secret
  client_id           = data.azurerm_client_config.current.client_id
  tenant_id           = data.azurerm_client_config.current.tenant_id
}
