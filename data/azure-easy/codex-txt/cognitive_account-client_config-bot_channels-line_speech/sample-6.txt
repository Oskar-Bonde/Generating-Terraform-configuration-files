terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "rg-terraform-demo"
  location = "eastus"
}

# create cognitive account resource
resource "azurerm_cognitive_account" "cog" {
  name                = "cog-terraform-demo"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  kind                = "TextAnalytics"
  sku_name            = "S0"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# create a service principal
resource "azurerm_azuread_application" "app" {
  name = "app-terraform-demo"
}

# create bot channels registration resource
resource "azurerm_bot_channels_registration" "bot" {
  name                = "bot-terraform-demo"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku                 = "F0"
  kind                = "sdk"
  endpoint            = "https://bot-terraform-demo.azurewebsites.net/api/messages"
  microsoft_app_id    = azurerm_azuread_application.app.application_id
  developer_app_insights_key = "00000000-0000-0000-0000-000000000000"
  developer_app_insights_api_key = "00000000-0000-0000-0000-000000000000"
  developer_app_insights_application_id = "00000000-0000-0000-0000-000000000000"
  tags = {
    environment = "Terraform Demo"
  }
}

# resource block called bot channel direct line speech
resource "azurerm_bot_channel_directline_speech" "bot" {
  name                = "bot-terraform-demo"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  bot_name            = azurerm_bot_channels_registration.bot.name
  cognitive_models    = [
    {
      id   = "00000000-0000-0000-0000-000000000000"
      name = "00000000-0000-0000-0000-000000000000"
    },
  ]
  speech_subscription_key = azurerm_cognitive_account.cog.primary_access_key
  speech_region           = azurerm_cognitive_account.cog.location
  speech_endpoint_id      = azurerm_cognitive_account.cog.endpoint
  client_id               = data.azurerm_client_config.current.client_id
  client_secret           = data.azurerm_client_config.current.client_secret
  tenant_id               = data.azurerm_client_config.current.tenant_id
}
