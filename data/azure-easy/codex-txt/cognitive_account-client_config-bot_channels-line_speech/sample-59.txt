terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "rg-name"
  location = "westus"
}

# create cognitive account resource
resource "azurerm_cognitive_account" "cog" {
  name                = "cog-name"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku_name            = "S0"
  kind                = "TextAnalytics"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# create cognitive account key
resource "azurerm_cognitive_account_key" "cog_key" {
  name                = azurerm_cognitive_account.cog.name
  resource_group_name = azurerm_resource_group.rg.name
  key_name            = "key1"
}

# create bot channels registration resource
resource "azurerm_bot_channels_registration" "bot" {
  name                = "bot-name"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku_name            = "F0"
  kind                = "sdk"
  microsoft_app_id    = "app-id"
  microsoft_app_password = "app-password"
}

# resource block called bot channel direct line speech
resource "azurerm_bot_channel_direct_line_speech" "bot_channel" {
  name                = "bot-channel-name"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  bot_name            = azurerm_bot_channels_registration.bot.name
  cognitive_models    = [
    {
      locale = "en-us"
      model_id = "model-id"
    }
  ]
  speech_subscription_key = azurerm_cognitive_account_key.cog_key.key
  speech_region = "westus"
  speech_context = "speech-context"
  speech_endpoint_id = "speech-endpoint-id"
  speech_authorization_token = data.azurerm_client_config.current.object_id
}
