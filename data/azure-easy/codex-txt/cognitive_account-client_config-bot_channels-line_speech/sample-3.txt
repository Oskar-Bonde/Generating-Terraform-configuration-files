terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "rg-test"
  location = "westus"
}

# create cognitive account resource
resource "azurerm_cognitive_account" "cog" {
  name                = "cog-test"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  kind                = "TextAnalytics"
  sku_name            = "S0"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# data block, azurem cognitive account keys
data "azurerm_cognitive_account_keys" "cog" {
  name                = azurerm_cognitive_account.cog.name
  resource_group_name = azurerm_resource_group.rg.name
}

# create bot channels registration resource
resource "azurerm_bot_channels_registration" "bot" {
  name                = "bot-test"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku                 = "F0"
  kind                = "sdk"
  endpoint            = "https://bot-test.azurewebsites.net/api/messages"
  msa_app_id          = data.azurerm_client_config.current.object_id
  msa_password        = data.azurerm_client_config.current.client_secret
  developer_app_insights_key = "12345678-1234-1234-1234-123456789012"
  developer_app_insights_api_key = "12345678-1234-1234-1234-123456789012"
  developer_app_insights_application_id = "12345678-1234-1234-1234-123456789012"
  cognitive_models {
    qna {
      name = "qna-test"
      subscription_key = data.azurerm_cognitive_account_keys.cog.key1
      knowledge_base_id = "12345678-1234-1234-1234-123456789012"
    }
  }
}

# resource block called bot channel direct line speech
resource "azurerm_bot_channel_direct_line_speech" "bot" {
  name                = "bot-test"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  bot_id              = azurerm_bot_channels_registration.bot.id
  subscription_key    = data.azurerm_cognitive_account_keys.cog.key1
  region              = "westus"
}
