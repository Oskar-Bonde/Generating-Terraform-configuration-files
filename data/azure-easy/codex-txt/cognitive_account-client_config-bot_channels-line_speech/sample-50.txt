terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "rg-test"
  location = "eastus"
}

# create cognitive account resource
resource "azurerm_cognitive_account" "cog" {
  name                = "cog-test"
  resource_group_name = azurerm_resource_group.rg.name
  location            = azurerm_resource_group.rg.location
  kind                = "TextAnalytics"
  sku_name            = "S0"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# data block, azurem cognitive account
data "azurerm_cognitive_account" "cog" {
  name                = azurerm_cognitive_account.cog.name
  resource_group_name = azurerm_resource_group.rg.name
}

# create bot channels registration resource
resource "azurerm_bot_channels_registration" "bot" {
  name                = "bot-test"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku                 = "F0"
  microsoft_app_id    = data.azurerm_client_config.current.object_id
  cognitive_models {
    cognitive_model_id = data.azurerm_cognitive_account.cog.id
    cognitive_model_name = data.azurerm_cognitive_account.cog.name
  }
}

# resource block called bot channel direct line speech
resource "azurerm_bot_channel_directline_speech" "bot" {
  name                = "bot-test"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  bot_name            = azurerm_bot_channels_registration.bot.name
  app_id              = azurerm_bot_channels_registration.bot.microsoft_app_id
  app_password        = azurerm_bot_channels_registration.bot.microsoft_app_password
  endpoint            = azurerm_bot_channels_registration.bot.endpoint
  speech_subscription_key = azurerm_bot_channels_registration.bot.speech_subscription_key
  speech_region       = azurerm_bot_channels_registration.bot.speech_region
}
