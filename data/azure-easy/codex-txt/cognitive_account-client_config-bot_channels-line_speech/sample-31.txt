terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "myResourceGroup"
  location = "West US"
}

# create cognitive account resource
resource "azurerm_cognitive_account" "cog_account" {
  name                = "myCognitiveAccount"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  kind                = "TextAnalytics"
  sku_name            = "S0"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# data block, azurem cognitive account keys
data "azurerm_cognitive_account_keys" "cog_account_keys" {
  name                = azurerm_cognitive_account.cog_account.name
  resource_group_name = azurerm_resource_group.rg.name
}

# create bot channels registration resource
resource "azurerm_bot_channels_registration" "bot_channels_registration" {
  name                = "myBotChannelsRegistration"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku                 = "F0"
  microsoft_app_id    = "myMicrosoftAppId"
  microsoft_app_password = "myMicrosoftAppPassword"
  endpoint            = "https://myBotChannelsRegistration.azurewebsites.net/api/messages"
  tags                = {
    environment = "staging"
  }
}

# resource block called bot channel direct line speech
resource "azurerm_bot_channel_direct_line_speech" "bot_channel_direct_line_speech" {
  name                = "myBotChannelDirectLineSpeech"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  bot_channels_registration_id = azurerm_bot_channels_registration.bot_channels_registration.id
  cognitive_services_authoring_key = data.azurerm_cognitive_account_keys.cog_account_keys.key1
  cognitive_services_authoring_region = "westus"
  speech_subscription_key = "mySpeechSubscriptionKey"
  speech_region = "westus"
}
