terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "my-rg"
  location = "West US"
}

# create cognitive account resource
resource "azurerm_cognitive_account" "cog_acc" {
  name                = "my-cog-acc"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku_name            = "S0"
  kind                = "TextAnalytics"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# create cognitive key resource
resource "azurerm_cognitive_account_key" "cog_key" {
  name                = azurerm_cognitive_account.cog_acc.name
  resource_group_name = azurerm_resource_group.rg.name
  key_name            = "key1"
  key_vault_uri       = "https://my-key-vault.vault.azure.net/"
  key_vault_client_id = data.azurerm_client_config.current.client_id
  key_vault_client_secret = data.azurerm_client_config.current.client_secret
}

# create bot channels registration resource
resource "azurerm_bot_channels_registration" "bot_reg" {
  name                = "my-bot-reg"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku_name            = "F0"
  microsoft_app_id    = "my-app-id"
  microsoft_app_password = "my-app-password"
}

# resource block called bot channel direct line speech
resource "azurerm_bot_channel_direct_line_speech" "bot_channel_direct_line_speech" {
  name                = "my-bot-channel-direct-line-speech"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  bot_channels_registration_name = azurerm_bot_channels_registration.bot_reg.name
  cognitive_services_account_name = azurerm_cognitive_account.cog_acc.name
  cognitive_services_account_key = azurerm_cognitive_account_key.cog_key.key
  cognitive_services_account_region = azurerm_cognitive_account.cog_acc.location
}
