terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure
provider "azurerm" {
  version = "=2.0.0"
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "my-rg"
  location = "westus"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# data block, azurem subscription
data "azurerm_subscription" "current" {}

# data block, azurem tenant
data "azurerm_tenant" "current" {}

# create aks cluster
resource "azurerm_kubernetes_cluster" "aks" {
  name                = "my-aks"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  dns_prefix          = "my-aks"

  default_node_pool {
    name       = "default"
    node_count = 1
    vm_size    = "Standard_DS2_v2"
  }

  service_principal {
    client_id     = data.azurerm_client_config.current.service_principal_client_id
    client_secret = data.azurerm_client_config.current.service_principal_client_secret
  }

  role_based_access_control {
    enabled = true
  }
}

# create bot channels registration resource
resource "azurerm_bot_channels_registration" "bot" {
  name                = "my-bot"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku                 = "F0"
  kind                = "bot"
  microsoft_app_id    = "00000000-0000-0000-0000-000000000000"
  microsoft_app_password = "my-bot-password"
}
