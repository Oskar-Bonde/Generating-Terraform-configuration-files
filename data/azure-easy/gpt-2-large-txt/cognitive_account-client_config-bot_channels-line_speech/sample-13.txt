terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure provider

provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg_policy" {
  name     = "myTFResourceGrouppolicy"
  location = "westus2"
}

# create cognitive account resource
resource "azurerm_storage_account" "rg_storage_account" {
  name                = "myTFResourceGroup"
  resource_group_name = azurerm_resource_group.rg.name
  location            = "westus2"
  account_tier             = "Standard"
  account_replication_type = "LRS"
}

# data block, azurem client config
resource "azurerm_storage_container" "rg_storage_container" {
  name                  = "myTFResourceGroup"
  resource_group_name   = azurerm_resource_group.rg.name
  storage_account_name  = azurerm_storage_account.rg_storage_account.name
  container_access_type = "private"
}

# create bot channels registration resource
resource "azurerm_storage_container" "rg_storage_container_linux" {
  name                  = "myTFResourceGroup"
  resource_group_name   = azurerm_resource_group.rg.name
  storage_account_name  = azurerm_storage_account.rg_storage_account.name
  container_access_type = "private"
}

# resource block called bot channel direct line speech
resource "azurerm_storage_container" "rg_storage_container_linux_linux" {
  name                  = "myTFResourceGroup"
  resource_group_name   = azurerm_resource_group.rg.name
  storage_account_name  = azurerm_storage_account.rg_storage_account.name
  container_access_type = "private"
}
