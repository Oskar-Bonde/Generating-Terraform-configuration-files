terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure provider

provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg_policy" {
  name     = "myTFResourceGroup_policy"
  location = "westus2"
}

# create cognitive account resource
resource "azurerm_storage_account" "rg" {
  name                = "myTFResourceGroup_storage_account"
  resource_group_name = azurerm_resource_group.rg.name
  location            = "westus2"
}

# data block, azurem client config
resource "azurerm_storage_container" "rg" {
  name                  = "myTFResourceGroup_storage_container"
  resource_group_name   = azurerm_resource_group.rg.name
  storage_account_name  = azurerm_storage_account.rg.name
  container_access_type = "private"
}

# create bot channels registration resource
resource "azurerm_storage_container" "rg_policy" {
  name                  = "myTFResourceGroup_storage_container_policy"
  resource_group_name   = azurerm_resource_group.rg.name
  storage_account_name  = azurerm_storage_account.rg.name
  container_access_type = "private"
}

# resource block called bot channel direct line speech
resource "azurerm_storage_container" "rg_policy_lb" {
  name                  = "myTFResourceGroup_storage_container_policy_lb"
  resource_group_name   = azurerm_resource_group.rg.name
  storage_account_name  = azurerm_storage_account.rg.name
  container_access_type = "private"
}
