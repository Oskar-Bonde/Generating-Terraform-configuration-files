terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "rg-terraform-demo"
  location = "West Europe"
}

# create cognitive account resource
resource "azurerm_cognitive_account" "cog_account" {
  name                = "cog-account-demo"
  resource_group_name = azurerm_resource_group.rg.name
  location            = azurerm_resource_group.rg.location
  kind                = "TextAnalytics"
  sku_name            = "S0"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# create bot channels registration resource
resource "azurerm_bot_channels_registration" "bot_registration" {
  name                = "bot-registration-demo"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku_name            = "F0"
  microsoft_app_id    = "12345678-1234-1234-1234-123456789012"
  microsoft_app_password = "12345678-1234-1234-1234-123456789012"
  endpoint            = "https://my-bot-app.azurewebsites.net/api/messages"
  tags                = {
    environment = "Terraform Demo"
  }
}
