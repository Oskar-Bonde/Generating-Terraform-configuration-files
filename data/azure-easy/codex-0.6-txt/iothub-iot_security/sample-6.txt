terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "my-resource-group"
  location = "West US"
}

# make iothub resource
resource "azurerm_iothub" "iothub" {
  name                = "my-iothub"
  resource_group_name = azurerm_resource_group.rg.name
  location            = azurerm_resource_group.rg.location
  sku                 = "S1"
}

# iot security solution resource
resource "azurerm_iothub_security_solution" "iot-security-solution" {
  iothub_name                = azurerm_iothub.iothub.name
  resource_group_name        = azurerm_resource_group.rg.name
  location                   = azurerm_resource_group.rg.location
  workspace_resource_id      = "subscriptions/00000000-0000-0000-0000-000000000000/resourceGroups/my-resource-group/providers/Microsoft.OperationalInsights/workspaces/my-workspace"
  workspace_subscription_id  = "00000000-0000-0000-0000-000000000000"
  workspace_tenant_id        = "00000000-0000-0000-0000-000000000000"
}
