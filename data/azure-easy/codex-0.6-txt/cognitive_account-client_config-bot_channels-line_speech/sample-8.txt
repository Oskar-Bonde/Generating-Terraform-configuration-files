terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "my-resource-group"
  location = "westus"
}

# create cognitive account resource
resource "azurerm_cognitive_account" "cog" {
  name                = "my-cognitive-account"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  kind                = "TextAnalytics"
  sku_name            = "S0"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# create bot channels registration resource
resource "azurerm_bot_channels_registration" "bot" {
  name                = "my-bot-channels-registration"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku_name            = "F0"
  microsoft_app_id    = data.azurerm_client_config.current.object_id
  developer_app_insights_key = "00000000-0000-0000-0000-000000000000"
  developer_app_insights_api_key = "00000000-0000-0000-0000-000000000000"
  cognitive_models {
    qna {
      name = "QnA"
      subscription_key = "00000000-0000-0000-0000-000000000000"
      knowledge_base_id = "00000000-0000-0000-0000-000000000000"
    }
  }
}

# resource block called bot channel direct line speech
resource "azurerm_bot_channel_direct_line_speech" "bot" {
  name                = "my-bot-channel-direct-line-speech"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  bot_name            = azurerm_bot_channels_registration.bot.name
  bot_resource_group  = azurerm_bot_channels_registration.bot.resource_group_name
  cognitive_services_account_name = azurerm_cognitive_account.cog.name
  cognitive_services_account_resource_group = azurerm_cognitive_account.cog.resource_group_name
  cognitive_services_account_location = azurerm_cognitive_account.cog.location
}
