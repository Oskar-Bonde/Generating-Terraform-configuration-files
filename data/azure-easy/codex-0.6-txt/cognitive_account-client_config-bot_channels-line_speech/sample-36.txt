terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure
provider "azurerm" {
  version = "=2.0.0"
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "rg-terraform-demo"
  location = "eastus"
}

# create cognitive account resource
resource "azurerm_cognitive_account" "cog_acc" {
  name                = "cogaccdemo"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  kind                = "TextAnalytics"
  sku_name            = "S0"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# create bot channels registration resource
resource "azurerm_bot_channels_registration" "bot" {
  name                = "botdemo"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku_name            = "F0"
  microsoft_app_id    = "appid"
  microsoft_app_password = "apppassword"
}

# resource block called bot channel direct line speech
resource "azurerm_bot_channel_direct_line_speech" "bot_channel_direct_line_speech" {
  name                = "botchanneldirectlinespeechdemo"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  bot_name            = azurerm_bot_channels_registration.bot.name
  bot_resource_group  = azurerm_bot_channels_registration.bot.resource_group_name
  cognitive_services_authoring_key = azurerm_cognitive_account.cog_acc.primary_access_key
  cognitive_services_authoring_region = azurerm_cognitive_account.cog_acc.location
  speech_subscription_key = data.azurerm_client_config.current.subscription_id
  speech_region = azurerm_cognitive_account.cog_acc.location
}
