terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "myResourceGroup"
  location = "West US"
}

# create cognitive account resource
resource "azurerm_cognitive_account" "cog" {
  name                = "mycognitiveaccount"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku_name            = "S0"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# create bot channels registration resource
resource "azurerm_bot_channels_registration" "bot" {
  name                = "mybot"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku                 = "F0"
  kind                = "bot"
  microsoft_app_id    = azurerm_cognitive_account.cog.app_id
  microsoft_app_password = azurerm_cognitive_account.cog.endpoint_key
  endpoint            = "https://${azurerm_bot_channels_registration.bot.name}.azurewebsites.net/api/messages"
  developer_app_insights_key = azurerm_cognitive_account.cog.app_insights_key
  developer_app_insights_api_key = azurerm_cognitive_account.cog.app_insights_api_key
  developer_app_insights_application_id = azurerm_cognitive_account.cog.app_insights_application_id
  tags = {
    environment = "production"
  }
}

# resource block called bot channel direct line speech
resource "azurerm_bot_channel_direct_line_speech" "bot" {
  name                = "mybot"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  bot_channels_registration_id = azurerm_bot_channels_registration.bot.id
  client_secret       = azurerm_cognitive_account.cog.endpoint_key
  client_id           = azurerm_cognitive_account.cog.app_id
  speech_subscription_key = azurerm_cognitive_account.cog.speech_subscription_key
  speech_region       = azurerm_cognitive_account.cog.speech_region
  tags = {
    environment = "production"
  }
}
