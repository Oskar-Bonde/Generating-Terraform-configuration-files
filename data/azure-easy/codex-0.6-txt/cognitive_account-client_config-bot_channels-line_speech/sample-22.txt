terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azure
provider "azurerm" {
  features {}
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "rg-name"
  location = "eastus"
}

# create cognitive account resource
resource "azurerm_cognitive_account" "cog" {
  name                = "cog-name"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  kind                = "TextAnalytics"
  sku_name            = "S0"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# create bot channels registration resource
resource "azurerm_bot_channels_registration" "bot" {
  name                = "bot-name"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  sku                 = "F0"
  microsoft_app_id    = "app-id"
  microsoft_app_password = "app-password"
  endpoint            = "https://bot-name.azurewebsites.net/api/messages"
  developer_app_insights_key = "app-insights-key"
  developer_app_insights_api_key = "app-insights-api-key"
  cognitive_models {
    dispatch_model_id = azurerm_cognitive_account.cog.id
    version = "0.1"
  }
  tags = {
    environment = "dev"
  }
}

# resource block called bot channel direct line speech
resource "azurerm_bot_channel_direct_line_speech" "bot" {
  name                = "bot-name"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  bot_name            = azurerm_bot_channels_registration.bot.name
  bot_resource_group  = azurerm_bot_channels_registration.bot.resource_group_name
  client_id           = data.azurerm_client_config.current.client_id
  client_secret       = data.azurerm_client_config.current.client_secret
  tenant_id           = data.azurerm_client_config.current.tenant_id
  speech_subscription_key = "speech-subscription-key"
  speech_region       = "speech-region"
}
