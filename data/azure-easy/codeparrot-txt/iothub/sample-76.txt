terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azurerm resource group
resource "azurerm_resource_group" "rg" {
  name     = "myTFResourceGroup"
  location = "westus2"
}

# create resource group policy
resource "azurerm_resource_group_policy" "policy" {
  name     = "myTFResourceGroupPolicy"
  location = "westus2"
  resource_group_name = azurerm_resource_group.rg.name

  # allow outbound tcp traffic on all ports
  egress {
    from_port = 0
    to_port = 65535
    protocol = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }

  # allow outbound udp traffic on a port range
  egress {
    from_port = 0
    to_port = 65535
    protocol = "udp"
    cidr_blocks = ["0.0.0.0/0"]
  }

  # allow outbound icmp traffic of a specific type
  egress {
    from_port = -1
    to_port = -1
    protocol = "icmp"
    cidr_blocks = ["0.0.0.0/0"]
  }
}

# iothub resource block
resource "azurerm_resource_group" "iothub" {
  name     = "myTFResourceGroup"
  location = "westus2"
}
