terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azurerm resource group
resource "azurerm_resource_group" "rg" {
  name     = "myTFRG"
  location = "westus2"
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "myTFRG"
  location = "westus2"
}

# create cognitive account resourceter
resource "azurerm_cosmosdb_account" "cognitive" {
  name                = "myCognitiveCognitive"
  resource_group_name = azurerm_resource_group.rg.name
  location            = "westus2"
  account_type        = "Standard_LRS"
}

# data block, azurem client config
data "template_file" "user_data_script" {
  template = file("${path.module}/user-data-script.sh")

  vars = {
    # terraform version
    terraform_version = var.terraform_version
  }
}

# create bot channels registration resource
resource "azurerm_bot_channel" "channel" {
  name                = "myBotChannel"
  resource_group_name = azurerm_resource_group.rg.name
  location            = "westus2"
  channel_role        = "channelRole-example"
}
