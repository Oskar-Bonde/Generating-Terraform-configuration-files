terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azurerm resource group
resource "azurerm_resource_group" "rg" {
  name     = "myTFRG"
  location = "westus2"
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "myTFRG"
  location = "westus2"
}

# make iothub resourcevariable
resource "azurerm_resource_variable" "resource_group_name" {
  name = "myTFResourceGroup"
  type = "Azure::Iot::Default"
}

# iot security solution resource
resource "azurerm_resource_group_security_solution" "iot_security_solution" {
  name                = "myiotSSSSSolution"
  location            = "westus2"
  resource_group_name = "${azurerm_resource_group.rg.name}"
  security_solution_name = "myiotSSSSSolution"
}
