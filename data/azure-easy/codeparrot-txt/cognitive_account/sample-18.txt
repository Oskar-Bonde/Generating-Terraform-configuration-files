terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azurerm resource group
resource "azurerm_resource_group" "rg" {
  name     = "myTFRG"
  location = "westus2"
}

# make resource group private to the azurerm resource group
resource "azurerm_resource_group" "private_rg" {
  name     = "myTFRG"
  location = "westus2"
}

# create cognitive account resource group
resource "azurerm_resource_group" "cognitive_rg" {
  name     = "myTFRG"
  location = "westus2"
}
