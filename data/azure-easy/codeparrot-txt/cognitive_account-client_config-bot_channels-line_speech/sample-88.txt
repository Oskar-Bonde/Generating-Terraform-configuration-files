terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azurerm resource group
resource "azurerm_resource_group" "rg" {
  name     = "myTFResourceGroup"
  location = "westus2"
}

# create resource group with required parameters
resource "azurerm_resource_group" "rg" {
  name     = "myTFResourceGroup"
  location = "westus2"
  resource_group_name = azurerm_resource_group.rg.name
}

# create cognitive account resource group
resource "azurerm_resource_group" "cognitive_sa" {
  name     = "myTFCognitiveSA"
  location = "westus2"
}

# data block, azurem client config
data "azurerm_subscription" "current" {
  name                = "myTFSubscription"
  resource_group_name = azurerm_resource_group.rg.name
  location            = "westus2"
}

# create bot channels registration resource
resource "azurerm_bot_application_registration" "bot_registration" {
  name                = "myTFBotRegistration"
  resource_group_name = azurerm_resource_group.rg.name
  location            = "westus2"
  sku                 = "Standard"
}

# resource block called bot channel direct line speech
resource "azurerm_bot_application_chat_incoming_webhook" "bot_chat_incoming_webhook" {
  name                = "myTFBotChatIncomingWebhook"
  resource_group_name = azurerm_resource_group.rg.name
  location            = "westus2"
  sku                 = "Standard"
  integration_http_method = "POST"
  integration_type     = "WEBHOOK"
  integration_http_uri   = "https://hooks.azurerm.com/services/api/"
  integration_http_method_uri = "POST"
}
