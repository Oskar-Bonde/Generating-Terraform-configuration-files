terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azurerm resource group
resource "azurerm_resource_group" "rg" {
  name     = "myTFRG"
  location = "westus2"
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "myTFRG"
  location = "westus2"
}

# create cognitive account resource group
resource "azurerm_resource_group" "cognitive_rg" {
  name     = "myTFRG"
  location = "westus2"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# terraform block, azurem client config
resource "azurerm_client_config" "current" {
  name                = "myTFClientConfig"
  resource_group_name = azurerm_resource_group.rg.name
  location            = data.azurerm_resource_group.rg.location
  admin_enabled        = true
  sku                 = "Standard"
  tags                = {
    environment = "Development"
  }
}

# create bot channels registration resource
resource "azurerm_bot_application_registration" "bot_registration" {
  name                = "myTFBotRegistration"
  resource_group_name = azurerm_resource_group.rg.name
  location            = data.azurerm_resource_group.rg.location
  sku                 = "Standard"
  tags                = {
    environment = "Development"
  }
}

# resource block called bot channel direct line speech
resource "azurerm_bot_application_chat_incoming" "bot_chat_incoming" {
  name                = "myTFBotChatIncoming"
  resource_group_name = azurerm_resource_group.rg.name
  location            = data.azurerm_resource_group.rg.location
  sku                 = "Standard"
  tags                = {
    environment = "Development"
  }
}
