terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azurerm resource group
resource "azurerm_resource_group" "rg" {
  name     = "myTFRG"
  location = "westus2"
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "myTFRG"
  location = "westus2"
}

# data block, azurem client config file
data "azurerm_client_config" "current" {}

# terraform block, azurerm client config file
resource "azurerm_client_config" "current" {
  name                = "myTFClientConfig"
  resource_group_name = azurerm_resource_group.rg.name
}

# create bot channels registration resource
resource "azurerm_bot_channel_registration" "bot_channel_registration" {
  name                = "myTFTCPublisher"
  resource_group_name = azurerm_resource_group.rg.name
  channel_registration_name = "myTFTCPublisher"
}
