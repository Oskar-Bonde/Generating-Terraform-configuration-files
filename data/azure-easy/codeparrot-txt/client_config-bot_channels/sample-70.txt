terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with azurerm resource group
resource "azurerm_resource_group" "rg" {
  name     = "myTFResourceGroup"
  location = "westus2"
}

# create resource group
resource "azurerm_resource_group" "rg" {
  name     = "myTFRG"
  location = "westus2"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# terraform block, azurem client config
resource "azurerm_client_config" "current" {
  name                = "myTFClientConfig"
  resource_group_name = azurerm_resource_group.rg.name
}

# create bot channels registration resource
resource "azurerm_bot_channel_registration" "channel_registration" {
  name                = "myTFBotRegistration"
  resource_group_name = azurerm_resource_group.rg.name
  channel_registration_endpoint = "https://${azurerm_bot_channel_registration.channel_registration.name}/registration"
}
