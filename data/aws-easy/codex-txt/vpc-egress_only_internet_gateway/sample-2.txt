terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider block 
provider "aws" {
  region = "us-east-1"
}

# create vpc resource
resource "aws_vpc" "main" {
  cidr_block = "10.0.0.0/16"
}

# create an egress only internet gateway
resource "aws_egress_only_internet_gateway" "egress" {
  vpc_id = "${aws_vpc.main.id}"
}
