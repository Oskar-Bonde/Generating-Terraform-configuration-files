terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider Block
provider "aws" {
  region = "us-east-1"
}

# Create a VPC resource
resource "aws_vpc" "foo" {
  cidr_block = "10.1.0.0/16"
}

# Make a service discovery private dns namespace resource
resource "aws_service_discovery_private_dns_namespace" "foo" {
  name        = "tf-sd-ns-%d"
  description = "test"
  vpc         = "${aws_vpc.foo.id}"
}
