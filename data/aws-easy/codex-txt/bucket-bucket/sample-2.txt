terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider Block 
provider "aws" {
  region = "us-east-1"
}

# Create S3 Bucket Resource
resource "aws_s3_bucket" "bucket" {
  bucket = "terraform-tfstate-%s"
  acl    = "private"
}

# Create S3 Bucket 
resource "aws_s3_bucket_object" "object" {
  bucket = "${aws_s3_bucket.bucket.bucket}"
  key    = "terraform.tfstate"
  source = "%s"
}
