terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider AWS block
provider "aws" {
  region = "us-east-1"
}

# Create EC2 Instance 
resource "aws_instance" "my_instance" {
  ami           = "ami-0e55e373"
  instance_type = "t2.micro"
}

# Output block
output "my_instance_id" {
  value = "${aws_instance.my_instance.id}"
}
