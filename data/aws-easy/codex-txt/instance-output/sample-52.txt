terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider AWS block
provider "aws" {
  region = var.region
}

# Create EC2 Instance 
resource "aws_instance" "web" {
  ami           = var.ami
  instance_type = var.instance_type

  tags = {
    Name = "terraform-example"
  }
}

# Output block
output "public_ip" {
  value = aws_instance.web.public_ip
}
