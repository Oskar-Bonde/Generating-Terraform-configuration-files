terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider AWS block
provider "aws" {
  region = "us-east-1"
}

# Create EC2 Instance 
resource "aws_instance" "ec2_instance" {
  ami           = "ami-0bbe6b35405ecebdb"
  instance_type = "t2.micro"
  tags {
    Name = "HelloWorld"
  }
}

# Output block
output "instance_ip" {
  value = "${aws_instance.ec2_instance.public_ip}"
}
