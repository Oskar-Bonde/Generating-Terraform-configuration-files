terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider Block AWS
provider "aws" {
  region = "us-east-1"
}

# Create variable 
variable "name" {
  type = string
}

# variable block
variable "ami" {
  type = string
}

# make variable
variable "instance_type" {
  type = string
}

# create variable block
variable "key_name" {
  type = string
}

# Create Security Group
resource "aws_security_group" "web" {
  name        = "web"
  description = "Allow HTTP and SSH access"
  vpc_id      = "vpc-0c5f5f3d7f8a4a8a7"

  ingress {
    description = "HTTP from VPC"
    from_port   = 80
    to_port     = 80
    protocol    = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }

  ingress {
    description = "SSH from VPC"
    from_port   = 22
    to_port     = 22
    protocol    = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }

  egress {
    from_port   = 0
    to_port     = 0
    protocol    = "-1"
    cidr_blocks = ["0.0.0.0/0"]
  }

  tags = {
    Name = "web"
  }
}
