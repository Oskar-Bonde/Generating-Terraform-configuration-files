terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider Block AWS
provider "aws" {
  region = "us-east-1"
}

# Create variable 
variable "name" {
  type = string
}

# variable block
variable "instance_type" {
  type = string
  default = "t2.micro"
}

# make variable
variable "ami" {
  type = string
  default = "ami-0d5d9d301c853a04a"
}

# create variable block
variable "count" {
  type = number
  default = 1
}

# Create Security Group
resource "aws_security_group" "default" {
  name = "terraform-sg"
  description = "Allow SSH and HTTP traffic"
  vpc_id = "vpc-0a7d9d3f2a8c1d8e0"

  ingress {
    description = "SSH"
    from_port = 22
    to_port = 22
    protocol = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }

  ingress {
    description = "HTTP"
    from_port = 80
    to_port = 80
    protocol = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }

  egress {
    from_port = 0
    to_port = 0
    protocol = "-1"
    cidr_blocks = ["0.0.0.0/0"]
  }

  tags = {
    Name = "terraform-sg"
  }
}

# Create Security Group
resource "aws_security_group" "default" {
  name = "terraform-sg"
  description = "Allow SSH and HTTP traffic"
  vpc_id = "vpc-0a7d9d3f2a8c1d8e0"

  ingress {
    description = "SSH"
    from_port = 22
    to_port = 22
    protocol = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }

  ingress {
    description = "HTTP"
    from_port = 80
    to_port = 80
    protocol = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }

  egress {
    from_port = 0
    to_port = 0
    protocol = "-1"
    cidr_blocks = ["0.0.0.0/0"]
  }

  tags = {
    Name = "terraform-sg"

# Create EC2 Instance
resource "aws_instance" "web" {
  ami = var.ami
  instance_type = var.instance_type
  count = var.count
  security_groups = [aws_security_group.default.name]
  tags = {
    Name = "terraform-instance"
  }
}
