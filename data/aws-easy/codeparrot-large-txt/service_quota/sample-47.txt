terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider block with AWS
provider "aws" {
  profile = "default"
  region  = "us-east-1"
}

# create a service quota resource
resource "aws_service_quota" "demo" {
  provider = "aws.foo"
  name = "test-quota"
  resource = "test_resource"
  in_use = "test_resource"
}
