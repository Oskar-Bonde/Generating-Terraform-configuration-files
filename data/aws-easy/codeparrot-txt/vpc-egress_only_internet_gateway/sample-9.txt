terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider block with AWS
provider "aws" {
  profile = "default"
  region  = "us-east-1"
}

# create vpc resource
resource "aws_vpc" "vpc1" {
  cidr_block = "10.0.0.0/16"
}

# create an egress only internet gateway
resource "aws_internet_gateway" "igw1" {
  vpc_id = "${aws_vpc.vpc1.id}"
}
