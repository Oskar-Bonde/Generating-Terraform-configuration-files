provider
terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider Block with AWS
provider "aws" {
  profile = "default"
  region  = "us-east-1"
}

# Create S3 Bucket Resource for Module
resource "aws_s3_bucket" "module" {
  bucket        = "modules.${var.project_name}"
  force_destroy = true
  acl           = "private"
  tags          = var.tags
}

# Create S3 Bucket object
resource "aws_s3_bucket" "module_bucket" {
  bucket        = aws_s3_bucket.module.id
  force_destroy = true
  acl           = "private"
  tags          = var.tags
}
