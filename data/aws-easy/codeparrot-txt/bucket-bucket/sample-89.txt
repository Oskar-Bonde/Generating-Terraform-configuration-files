provider
terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider Block with AWS
provider "aws" {
  profile = "default"
  region  = "us-east-1"
}

# Create S3 Bucket Resource for Module
resource "aws_s3_bucket" "module" {
  bucket        = "digitalmarketplace-terraform-state-production"
  acl           = "private"
  force_destroy = true

  versioning {
    enabled = true
  }

  tags {
    Name = "digitalmarketplace-terraform-state-production"
  }
}

# Create S3 Bucket object for Module
resource "aws_s3_bucket" "module_object" {
  bucket        = aws_s3_bucket.module.id
  acl           = "private"
  force_destroy = true

  versioning {
    enabled = true
  }

  tags {
    Name = "digitalmarketplace-terraform-state-production"
  }
}
