provider
terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider Block with AWS
provider "aws" {
  profile = "default"
  region  = "us-east-1"
}

# Create S3 Bucket Resource for Module
resource "aws_s3_bucket" "module" {
  bucket = "nubis-terraform-state"
  acl    = "private"

  tags {
    Name = "nubis-terraform-state"
  }
}

# Create S3 Bucket object
resource "aws_s3_bucket" "state" {
  bucket = "nubis-terraform-state"
  acl    = "private"

  tags {
    Name = "nubis-terraform-state"
  }
}
