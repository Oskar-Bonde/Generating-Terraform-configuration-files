provider "aws" {
  profile = "default"
  region  = "us-east-1"
}

# AWS provider block
provider "aws" {
  alias  = "files"
  region = "us-east-1"
}

# Resource block to create a random pet name
resource "random_pet_name" "test" {
  byte_length = 2
}

# Create AWS S3 bucketvariable
resource "aws_s3_bucket_object" "test" {
  bucket = "${var.bucket_name}"
  key    = "test.txt"
  content = "${file("test.txt")}"
}
