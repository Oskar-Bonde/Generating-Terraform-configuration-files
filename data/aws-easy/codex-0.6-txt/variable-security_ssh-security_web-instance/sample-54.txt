terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider Block AWS
provider "aws" {
  region = "us-east-1"
}

# Create variable 
variable "name" {
  type = string
}

# variable block
variable "instance_type" {
  type = string
  default = "t2.micro"
}

# make variable
variable "ami" {
  type = string
  default = "ami-0bbe6b35405ecebdb"
}

# create variable block
variable "count" {
  type = number
  default = 1
}

# Create Security Group
resource "aws_security_group" "sg" {
  name = "terraform-sg"
  description = "terraform security group"
  vpc_id = "vpc-0d3f4c4d9c9f9e2c4"

  ingress {
    from_port = 22
    to_port = 22
    protocol = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }

  egress {
    from_port = 0
    to_port = 0
    protocol = "-1"
    cidr_blocks = ["0.0.0.0/0"]
  }
}

# Create Security Group
resource "aws_security_group" "sg2" {
  name = "terraform-sg2"
  description = "terraform security group2"
  vpc_id = "vpc-0d3f4c4d9c9f9e2c4"

  ingress {
    from_port = 80
    to_port = 80
    protocol = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }

  egress {
    from_port = 0
    to_port = 0
    protocol = "-1"
    cidr_blocks = ["0.0.0.0/0"]
  }
}

# Create EC2 Instance
resource "aws_instance" "web" {
  ami = var.ami
  instance_type = var.instance_type
  count = var.count
  security_groups = [aws_security_group.sg.name, aws_security_group.sg2.name]
  tags = {
    Name = "terraform-instance-${var.name}"
  }
}
