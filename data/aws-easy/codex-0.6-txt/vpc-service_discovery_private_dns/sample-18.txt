terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider Block
provider "aws" {
  region = "us-east-1"
}

# Create a VPC resource
resource "aws_vpc" "myvpc" {
  cidr_block = "10.0.0.0/16"
  tags = {
    Name = "myvpc"
  }
}

# Make a service discovery private dns namespace resource
resource "aws_service_discovery_private_dns_namespace" "my_private_dns_namespace" {
  name = "my_private_dns_namespace"
  vpc = aws_vpc.myvpc.id
}
