terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider Block 
provider "aws" {
  region = "us-east-1"
}

# Create S3 Bucket Resource
resource "aws_s3_bucket" "s3-bucket" {
  bucket = "terraform-s3-bucket-demo"
  acl    = "private"
}

# Create S3 Bucket object
resource "aws_s3_bucket_object" "s3-bucket-object" {
  bucket = "${aws_s3_bucket.s3-bucket.id}"
  key    = "terraform-s3-bucket-object-demo"
  source = "./test.txt"
}
