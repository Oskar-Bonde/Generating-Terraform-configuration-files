terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider Block 
provider "aws" {
  region = "us-east-1"
}

# Create S3 Bucket Resource
resource "aws_s3_bucket" "my-s3-bucket" {
  bucket = "my-s3-bucket"
  acl    = "private"
}

# Create S3 Bucket object
resource "aws_s3_bucket_object" "my-s3-bucket-object" {
  bucket = aws_s3_bucket.my-s3-bucket.id
  key    = "my-s3-bucket-object"
  source = "my-s3-bucket-object.txt"
}
