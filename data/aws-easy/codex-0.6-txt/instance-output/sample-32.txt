terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider AWS block
provider "aws" {
  region = "us-east-1"
}

# Create EC2 Instance 
resource "aws_instance" "example" {
  ami           = "ami-0ac019f4fcb7cb7e6"
  instance_type = "t2.micro"
}

# Output block
output "public_ip" {
  value = aws_instance.example.public_ip
}
