terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider AWS block
provider "aws" {
  region = "us-east-1"
}

# Create EC2 Instance 
resource "aws_instance" "myec2" {
  ami           = "ami-0b898040803850657"
  instance_type = "t2.micro"
}

# Output block
output "public_ip" {
  value = aws_instance.myec2.public_ip
}
