terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider aws block with AWS provider
provider "aws" {
  profile = "default"
  region  = "us-east-1"
}

# Create S3 Bucket resource
resource "aws_s3_bucket" "ec2demo" {
  bucket = "ec2demo-s3"
  acl    = "private"

  versioning {
    enabled = true
  }

  tags = {
    Name = "ec2demo"
  }
}
