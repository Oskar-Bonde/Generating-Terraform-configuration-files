# Resource block to create a random pet namevariable
resource "random_pet" "random_name" {
  name       = "random_name"
  keepers = {
    random_name = "random_name"
  }
}

# Create AWS S3 bucket
resource "aws_s3_bucket" "random_name" {
  bucket = "${var.random_name}"
  acl    = "private"

  force_destroy = true

  versioning {
    enabled = true
  }

  lifecycle {
    ignore_changes = [
      acl                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                   
}
