terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider Block with AWS
provider "aws" {
  profile = "default"
  region  = "us-east-1"
}

# Create a AWS VPC resource
resource "aws_vpc" "main" {
  cidr_block = "10.0.0.0/16"
}

# make a vpn gateway to the VPC
resource "aws_internet_gateway" "gw" {
  vpc_id = "${aws_vpc.main.id}"
}

# create a customer gateway to the VPC
resource "aws_customer_gateway" "gw" {
  vpc_id = "${aws_vpc.main.id}"
}

# make a vpn connection
resource "aws_vpn_connection" "gw" {
  vpn_gateway_id = "${aws_vpn_gateway.gw.id}"
  customer_gateway_id = "${aws_customer_gateway.gw.id}"
}
