terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider block with AWS
provider "aws" {
  profile = "default"
  region  = "us-east-1"
}

# create a service quota resource
resource "aws_service_quota" "service_quota" {
  service  = "service"
  metric   = "service.quota"
  limit   = "10"
  period = "60"
}
