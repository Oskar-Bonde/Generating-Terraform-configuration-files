  required_providers {

      source  = "hashicorp/aws"
    }

  profile = "default"
  region  = "us-east-1"
}

  ami           = "ami-0ff8a91507f77f867"
  instance_type = "t2.micro"
}

# Output block
output "ec2instance" {
  value = "${aws_instance.ec2instance.id}"
}
