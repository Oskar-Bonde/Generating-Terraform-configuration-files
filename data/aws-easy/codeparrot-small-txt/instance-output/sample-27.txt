provider
terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider AWS block with AWS
provider "aws" {
  profile = "default"
  region  = "us-east-1"
}

# Create EC2 Instance
resource "aws_instance" "ec2demo" {
  ami           = "ami-0ff8a91507f77f867"
  instance_type = "t2.micro"
}

# Output blockresource for debug
# Terraform output is not compatible with any other AWS provider
# so we have to use the block() function to output the block()
output "ec2demo_ip" {
  value = "${aws_instance.ec2demo.public_ip}"
}
