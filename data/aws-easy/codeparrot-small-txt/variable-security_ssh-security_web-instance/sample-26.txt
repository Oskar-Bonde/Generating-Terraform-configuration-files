terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider Block AWS
provider "aws" {
  profile = "default"
  region  = "us-east-1"
}

# Create variable with default value
variable "vpc_cidr" {
  default = "10.0.0.0/16"
}

# variable block with default value
variable "subnet_cidr" {
  default = "10.0.1.0/24"
}

# make variable
variable "instance_type" {
  default = "t2.micro"
}

# create variable block
variable "user_data" {
  default = "#!/bin/bash\necho ECS_CLUSTER=cluster-${var.name} >> /etc/ecs/ecs.config"
}

# Create Security Group
resource "aws_security_group" "ecs_sg" {
  name        = "ecs_sg"
  description = "Security group for ${var.name} instances"
  vpc_id      = "${var.vpc_cidr}"

  ingress {
    from_port   = 0
    to_port     = 0
    protocol    = "-1"
    cidr_blocks = ["0.0.0.0/0"]
  }

  ingress {
    from_port   = 22
    to_port     = 22
    protocol    = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }

  ingress {
    from_port   = -1
    to_port     = -1
    protocol    = "icmp"
    cidr_blocks = ["0.0.0.0/0"]
  }

  egress {
    from_port   = 0
    to_port     = 0
    protocol    = "-1"
    cidr_blocks = ["0.0.0.0/0"]
  }

  tags {
    Name = "ecs_sg"
  }
}
