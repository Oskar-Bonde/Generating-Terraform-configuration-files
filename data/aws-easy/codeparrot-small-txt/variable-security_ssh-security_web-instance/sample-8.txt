terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider Block AWS
provider "aws" {
  profile = "default"
  region  = "us-east-1"
}

# Create variable with default value
variable "instance_name" {
  default = "terraform-example"
}

# variable block
variable "instance_type" {
  default = "t2.micro"
}

# make variable
variable "key_name" {
  default = "terraform-example"
}

# create variable blockvariable
variable "value" {
  default = "terraform-example"
}

# Create Security Group
resource "aws_security_group" "terraform-example" {
  name        = "${var.instance_name}-${var.key_name}"
  description = "Used in the terraform"
  vpc_id      = "${var.vpc_id}"

  # SSH access from anywhere
  ingress {
    from_port   = 22
    to_port     = 22
    protocol    = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }

  # HTTP access from anywhere
  ingress {
    from_port   = 80
    to_port     = 80
    protocol    = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }

  # outbound internet access
  egress {
    from_port   = 0
    to_port     = 0
    protocol    = "-1"
    cidr_blocks = ["0.0.0.0/0"]
  }
}

# Create Security Group
resource "aws_security_group" "terraform-example-sg" {
  name        = "${var.instance_name}-${var.key_name}"
  description = "Used in the terraform"
  vpc_id      = "${var.vpc
}
