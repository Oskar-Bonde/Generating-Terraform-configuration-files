terraform {
  required_providers {
    random = {
      source = "hashicorp/random"
    }
  }
}

# AWS provider block
provider "aws" {
  profile = "default"
  region  = "us-east-1"
}

# Resource block to create a random pet nameresource
resource "random_pet" "nameresource" {
  length    = 1
  prefix    = "terraform-test-${random_pet.nameresource.id}"
  separator = "-"
}

# Create AWS S3 bucket for Terraform state
resource "aws_s3_bucket" "terraform_state" {
  bucket = "terraform-test-${random_pet.nameresource.id}"
  acl    = "private"
}
