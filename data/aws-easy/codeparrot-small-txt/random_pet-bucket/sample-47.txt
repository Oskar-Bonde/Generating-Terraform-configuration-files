terraform {
  required_providers {
    random = {
      source = "hashicorp/random"
    }
  }
}

# AWS provider block
provider "aws" {
  profile = "default"
  region  = "us-east-1"
}

# Resource block to create a random pet nameresource
resource "random_pet" "pet" {
  length = 1
}

# Create AWS S3 bucketresource
resource "aws_s3_bucket" "pet" {
  bucket = "terraform-example-bucket"
  acl    = "private"
  force_destroy = true
}
