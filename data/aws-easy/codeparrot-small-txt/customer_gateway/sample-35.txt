terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider block with AWS
provider "aws" {
  profile = "default"
  region  = "us-east-1"
}

# create customer gateway
resource "aws_customer_gateway" "gw" {
  vpc_id = "${aws_vpc.default.id}"
  tags {
    Name = "customer-gateway"
  }
}
