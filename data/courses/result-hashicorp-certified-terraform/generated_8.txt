# Terraform Block with AWS provider
terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider AWS in region var.aws_region
provider "aws" {
  region = var.aws_region
}

# Create variable called aws_region with default "us-east-1"
variable "aws_region" {
  default = "us-east-1"
}

# Create variable called db_username that is sensitive
variable "db_username" {
  default = "dummy_user"
  sensitive = true
}

# create variable called db_password that is sensitive
variable "db_password" {
  default = "dummy_pass"
  sensitive = true
}

# Create RDS MySQL Database with 5 allocated storage and username var.db_username and password var.db_password
resource "aws_db_instance" "app" {
  allocated_storage = 5
  engine            = "mysql"
  engine_version    = "5.7"
  instance_class    = "db.t2.micro"
  name              = "mydb"
  username          = var.db_username
  password          = var.db_password
  parameter_group_name = "default.mysql5.7"
  skip_final_snapshot = true
}
