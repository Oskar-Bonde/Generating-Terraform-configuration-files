terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with features block
provider "azurerm" {
  features {}
}

# create resource group, use name example-resources and set location to West Europe
resource "azurerm_resource_group" "rg_ssh" {
  name     = "myTFResourceGroup-ssh"
  location = "westus2"
}

# create cognitive account resource. Of kind SpeechServices and sku name S0. Use name and location from resource group
resource "azurerm_storage_account" "rg_storage_account" {
  name                = "myTFResourceGroup-storage-account"
  resource_group_name = azurerm_resource_group.rg.name
  location            = "westus2"
  account_tier             = "Standard"
  account_replication_type = "LRS"
}

# data block, azurem client config
resource "azurerm_storage_container" "rg_storage_container" {
  name                  = "myTFResourceGroup-storage-container"
  resource_group_name   = azurerm_resource_group.rg.name
  storage_account_name  = azurerm_storage_account.rg_storage_account.name
  container_access_type = "private"
}

# create bot channels registration resource. Set location global, sku to F0, microsoft app id from client config id. Assign name and use resource group name
resource "azurerm_storage_container" "rg_storage_container_linux_container" {
  name                  = "myTFResourceGroup-storage-container-linux"
  resource_group_name   = azurerm_resource_group.rg.name
  storage_account_name  = azurerm_storage_account.rg_storage_account.name
  container_access_type = "private"
}
