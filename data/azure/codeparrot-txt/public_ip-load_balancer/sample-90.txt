terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with features block
provider "azurerm" {
  features {}
}

# create resource group, use name example in location West Europe
resource "azurerm_resource_group" "rg" {
  name     = "example"
  location = "westus"
}

# create public ip resource, use static allocation method. Assign resource group name and location
resource "azurerm_public_ip" "pip" {
  name                         = "pip.${var.name}-ip"
  location                     = "westus"
  resource_group_name          = azurerm_resource_group.rg.name
  allocation_method            = "Static"
}

# lb resource. In frontend ip configuration block, asssign name and public ip address id. Use resource group name and location to create public ip address.
resource "azurerm_lb" "lb" {
  name                = "lb.${var.name}-lb"
  location            = "westus"
  resource_group_name = azurerm_resource_group.rg.name
  sku                 = "Standard"
  allocation_method   = "Static"
  sku_name            = "Standard"
}
