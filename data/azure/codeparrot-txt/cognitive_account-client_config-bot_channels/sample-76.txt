terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with features block
provider "azurerm" {
  features {}
}

# create resource group, use name example-resources and set location to West Europe
resource "azurerm_resource_group" "example-resources" {
  name     = "example-resources"
  location = "westus"
}

# create cognitive account resource. Of kind SpeechServices and sku name S0. Use name and location from resource group
resource "azurerm_cosmosdb_account" "scott" {
  name                = "scott"
  resource_group_name = azurerm_resource_group.rg.name
  location            = "westus"
  account_type        = "Standard_LRS"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# create storage account
resource "azurerm_storage_account" "stor" {
  name                = "stor"
  resource_group_name = azurerm_resource_group.rg.name
  location            = "westus"
  account_type        = "Standard_LRS"
}

# create bot channels registration resource. Set location global, sku to F0, microsoft app id from client config id. Assign name and use resource group name
resource "azurerm_bot_application_registration" "tf_app_registration" {
  name                = "tf_app_registration"
  resource_group_name = azurerm_resource_group.rg.name
  location            = "westus"
  app_id              = azurerm_app_registration.tf_app_registration.id
  type                = "WebhookAppRegistration"
  webhook_endpoint     = "https://hooks.azure.com/subscriptions/${azurerm_subscription.tf_subscription.name}/resourceGroups/${azurerm_resource_group.rg.name}/providers/Microsoft.Web/sites/${azurerm_app_registration.tf_app_registration.name}/webhooks/${azurerm_app_registration.tf_app_registration.name}"
  token_uri        = "https://${azurerm_app_registration.tf_app_registration.name}.firebaseapp.com/oauth/token"
}
