terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with features block
provider "azurerm" {
  features {}
}

# create resource group, use name example in location West Europe
resource "azurerm_resource_group" "example" {
  name     = "example"
  location = "westus"
}

# iothub with sku name S1 and capacity 1. Use resource group name and location to create S1
resource "azurerm_storage_account" "stor" {
  name                     = "${var.account_name}"
  resource_group_name      = "${azurerm_resource_group.rg.name}"
  location                 = "${var.location}"
  account_tier             = "Standard"
  account_replication_type = "LRS"
}

# iot security solution, assign display name and name. Use iothub id and resrouce group name and location
resource "azurerm_storage_share" "iot" {
  name                 = "iot-storage-share"
  storage_account_name = "${azurerm_storage_account.stor.name}"
  quota                = 10
  depends_on            = ["azurerm_storage_account.stor"]
}
