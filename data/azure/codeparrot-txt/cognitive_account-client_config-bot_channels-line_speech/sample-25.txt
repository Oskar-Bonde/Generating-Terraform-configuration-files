terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with features block
provider "azurerm" {
  features {}
}

# create resource group, use name example-resources and set location to West Europe
resource "azurerm_resource_group" "example_resources" {
  name     = "example-resources"
  location = "westus"
}

# create cognitive account resource. Of kind SpeechServices and sku name S0. Use name and location from resource group.
resource "azurerm_cosmosdb_account" "cosmosdbaccount" {
  name                = "cosmosdbaccount"
  resource_group_name = azurerm_resource_group.rg.name
  location            = "westus"
}

# data block, azurem client config
data "template_file" "test_script" {
  template = file("${path.module}/files/script.sh")

  vars = {
    account_name = azurerm_cosmosdb_account.cosmosdbaccount.name
    account_key    = azurerm_cosmosdb_account.cosmosdbaccount.primary_access_key
    container_name = azurerm_resource_group.rg.name
    container_access_type = "private"
    container_access_id = azurerm_cosmosdb_account.cosmosdbaccount.primary_access_key
    container_password = azurerm_cosmosdb_account.cosmosdbaccount.primary_password
    container_storage_account_name = azurerm_cosmosdb_account.cosmosdbaccount.name
    container_container_name = azurerm_resource_group.rg.name
    container_access_type = "blob"
    container_access_id = azurerm_cosmosdb_account.cosmosdbaccount.primary_access_key
  }
}

# create bot channels registration resource. Set location global, sku to F0, microsoft app id from client config id. Assign name and use resource group name to get registration id.
resource "azurerm_bot_application_registration" "bot_registration" {
  name                = "bot_registration"
  location            = "westus"
  resource_group_name = azurerm_resource_group.rg.name
  app_id              = azurerm_bot_application_registration.bot_application_registration.app_id
  registration_id     = azurerm_bot_application_registration.bot_application_registration.registration_id
  sku                 = "West US"
}

# resource block called bot channel direct line speech. Use name and location from bot_channels_registration. For cognitive service location and access key use cognitive account. Assign resource group name to get registration id.
resource "azurerm_bot_application_registration" "bot_registration_direct_line_speech" {
  name                = "bot_registration_direct_line_speech"
  location            = "westus"
  resource_group_name = azurerm_resource_group.rg.name
  app_id              = azurerm_bot_application_registration.bot_application_registration.app_id
  registration_id     = azurerm_bot_application_registration.bot_application_registration.registration_id
  sku                 = "West US"
}
