terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with features block
provider "azurerm" {
  features {}
}

# create resource group, use name example-resources and set location to West Europe
resource "azurerm_resource_group" "example_resources" {
  name     = "example-resources"
  location = "westus"
}

# create cognitive account resource. Of kind SpeechServices and sku name S0. Use name and location from resource group
resource "azurerm_cosmosdb_account" "scott" {
  name                = "scott"
  resource_group_name = azurerm_resource_group.rg.name
  location            = "westus"
  account_type        = "Standard_LRS"
}

# data block, azurem client config
data "azurerm_client_config" "current" {}

# create storage account
resource "azurerm_storage_account" "stor" {
  name                = "stor"
  resource_group_name = azurerm_resource_group.rg.name
  location            = "westus"
  account_type        = "Standard_LRS"
}

# create bot channels registration resource. Set location global, sku to F0, microsoft app id from client config id. Assign name and use resource group name
resource "azurerm_bot_application_registration" "bot_registration" {
  name                = "bot_registration"
  resource_group_name = azurerm_resource_group.rg.name
  location            = "westus"
  sku                 = "global"
  app_id              = azurerm_app_service.bot.id
}

# resource block called bot channel direct line speech. Use name and location from bot_channels_registration. For cognitive service location and access key use cognitive account. Assign resource group name
resource "azurerm_bot_application_chat_incoming" "bot_chat_incoming" {
  name                = "bot_chat_incoming"
  resource_group_name = azurerm_resource_group.rg.name
  location            = "westus"
  chat_channel_id    = azurerm_bot_application_chat_incoming.bot_chat_incoming.id
  enabled             = true
  chat_username      = "chat_username"
  chat_service_name = "chat_service_name"
  chat_service_location = "westus"
  enable_chat_broadcast = true
  enable_chat_offline    = true
  enable_chat_logging    = true
  enable_chat_sms        = true
  enable_chat_sms_callback = true
  enable_chat_verify_credentials = true
  enable_chat_sync       = true
}
