terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
    }
  }
}

# Provider block with features block
provider "azurerm" {
  features {}
}

# create resource group, use name example-resources and set location to West Europe
resource "azurerm_resource_group" "example_resources" {
  name     = "example-resources"
  location = "westus"
}

# create cognitive account resource. Of kind SpeechServices and sku name S0. Use name and location from resource group.
resource "azurerm_cosmosdb_account" "tf_sa" {
  name                = "tf_sa"
  resource_group_name = azurerm_resource_group.rg.name
  location            = "westus"
  account_type        = "Standard_LRS"
}

# data block, azurem client config
data "template_file" "tf_sa_config" {
  template = file("${path.module}/templates/tf_sa_config.json")

  vars = {
    account_name = azurerm_cosmosdb_account.tf_sa.name
    account_key = azurerm_cosmosdb_account.tf_sa.primary_access_key
    account_type = "Standard_LRS"
    account_endpoint = azurerm_cosmosdb_account.tf_sa.primary_access_key
    container_name = azurerm_resource_group.example_resources.name
    container_access_type = "private"
    container_endpoint = azurerm_cosmosdb_account.tf_sa.primary_access_key
    container_access_config {
      name = "config"
    }
  }
}

# create bot channels registration resource. Set location global, sku to F0, microsoft app id from client config id. Assign name and use resource group name to get registration id.
resource "azurerm_bot_application_registration" "tf_sa_registration" {
  name                = "tf_sa_registration"
  location            = "westus"
  resource_group_name = azurerm_resource_group.rg.name
  app_id             = azurerm_bot_application_registration.tf_sa_registration.app_id
  registration_id     = azurerm_bot_application_registration.tf_sa_registration.id
  client_id           = azurerm_bot_application_registration.tf_sa_registration.client_id
  client_secret       = azurerm_bot_application_registration.tf_sa_registration.client_secret
  allowed_audiences     = ["https://cloud.google.com/cognitive-services/docs/cognitive-services-guide#real-time-and-privacy-policy-based-access-control"]
  allowed_device_identities = ["identity.cognitive.microsoft.com"]
}

# resource block called bot channel direct line speech. Use name and location from bot_channels_registration. For cognitive service location and access key use cognitive account. Assign resource group name to get registration id.
resource "azurerm_bot_channel" "tf_sa_channel" {
  name                = "tf_sa_channel"
  resource_group_name = azurerm_resource_group.rg.name
  location            = "westus"
  channel_access_type = "channel"
  depends_on          = [azurerm_bot_application_registration.tf_sa_registration]
}
