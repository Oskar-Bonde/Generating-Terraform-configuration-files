# raw-data/hashicorp-certified-terraform\03-Terraform-Fundamental-Blocks\03-03-Multiple-Provider-Configurations\terraform-manifests
# Terraform Block with the AWS provider
terraform {
  required_version = "~> 0.14.6"
  required_providers {
    aws = {
      source = "hashicorp/aws"
      version = "~> 3.0"
    }
  }
}

# Provider block with AWS in us-east-1 region and profile set as default
provider "aws" {
  region = "us-east-1"
  profile = "default"
}

# Provider block with AWS in us-west-1 region, profile set as default and alias aws-west-1
provider "aws" {
  region = "us-west-1"
  profile = "default"
  alias = "aws-west-1"
}

# Resource block to Create VPC called vpc-us-east-1 which uses private network 10.1.0.0/16
resource "aws_vpc" "vpc-us-east-1" {
  cidr_block = "10.1.0.0/16"
  tags = {
    "Name" = "vpc-us-east-1"
  }
}


# Resource block to Create VPC called vpc-us-west-1 which uses private network 10.1.0.0/16 the provider aws.aws-west-1
resource "aws_vpc" "vpc-us-west-1" {
  cidr_block = "10.1.0.0/16"
  provider = aws.aws-west-1
  tags = {
    "Name" = "vpc-us-west-1"
  }
}
