# Terraform Block with the aws provider
terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
  }
}

# Provider Block aws in us-east-1
provider "aws" {
  profile = "default"
  region  = "us-east-1"
}

# Resource block that creates an ec2 instance with ami-047a51fa27710816e, instance type t2.micro
resource "aws_instance" "ec2demo" {
  ami           = "ami-047a51fa27710816e"
  instance_type = "t2.micro"
}
