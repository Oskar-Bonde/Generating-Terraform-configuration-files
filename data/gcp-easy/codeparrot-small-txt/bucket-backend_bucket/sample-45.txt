terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block
provider "google" {
}

# create storage bucket resource
resource "google_storage_bucket" "my-bucket" {
  name = "my-bucket-${random_suffix}"
  location = "US"
}

# create compute backend bucketresource
resource "google_storage_bucket_object" "my-bucket-object" {
  name = "my-bucket-object-${random_suffix}"
  bucket = google_storage_bucket.my-bucket.name
  source = "hashicorp/google/latest/backend/gcs"
}
