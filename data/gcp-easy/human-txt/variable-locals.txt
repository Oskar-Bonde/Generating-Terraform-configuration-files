# Terraform block with google provider
terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block 
provider "google" {
}

# variable block
variable "name" {
  description = "A name of the service"
  type        = string
  default     = "service"
}

# local block
locals {
  common_labels = {
    service     = var.service_name
  }
}
