terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block 
provider "google" {
  credentials = file("${path.module}/credentials.json")
  project     = "my-project"
  region      = "us-central1"
}

# create a CaPool resource
resource "google_compute_region_autoscaler" "my-autoscaler" {
  name = "my-autoscaler"
  region = "us-central1"
  target = "https://www.googleapis.com/compute/v1/projects/my-project/zones/us-central1-a/instances/my-instance"
  autoscaling_policy {
    max_replicas = 5
    min_replicas = 1
    cooldown_period_sec = 60
    cpu_utilization {
      target = 0.5
    }
  }
}
