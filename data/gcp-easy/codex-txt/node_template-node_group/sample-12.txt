terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block
provider "google" {
  credentials = file("${var.credentials_file}")
  project     = "${var.project}"
  region      = "${var.region}"
}

# create a node template
resource "google_compute_instance_template" "node-template" {
  name           = "node-template"
  machine_type   = "n1-standard-1"
  can_ip_forward = false

  tags = ["node"]

  disk {
    source_image = "debian-cloud/debian-9"
    auto_delete  = true
    boot         = true
  }

  network_interface {
    network = "default"
  }

  metadata {
    ssh-keys = "ubuntu:${file("${var.public_key_file}")}"
  }

  service_account {
    scopes = ["https://www.googleapis.com/auth/cloud-platform"]
  }
}

# create a compute node group resource
resource "google_compute_node_group" "node-group" {
  name        = "node-group"
  description = "Node group for node-template"
  node_template = "${google_compute_instance_template.node-template.self_link}"
  target_size = 1
  zone        = "${var.zone}"
}
