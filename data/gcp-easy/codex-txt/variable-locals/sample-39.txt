terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block 
provider "google" {
  credentials = file("${var.credentials_file}")
  project     = "${var.project}"
  region      = "${var.region}"
}

# variable block
variable "credentials_file" {
  description = "Path to the service account json file"
}

# local block
locals {
  # local variable
  zone = "us-central1-a"
}
