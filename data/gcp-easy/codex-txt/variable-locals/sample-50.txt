terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block 
provider "google" {
  credentials = file("${var.credentials_file}")
  project     = "${var.project}"
  region      = "${var.region}"
}

# variable block
variable "credentials_file" {
  type = string
}

# local block
locals {
  # local variable
  my_local_variable = "my_local_variable_value"
}
