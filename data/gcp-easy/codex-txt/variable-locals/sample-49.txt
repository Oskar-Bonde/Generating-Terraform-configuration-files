terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block 
provider "google" {
  credentials = file("${path.module}/credentials.json")
  project     = var.project
  region      = var.region
}

# variable block
variable "project" {
  description = "The GCP project ID"
}

# local block
locals {
  project = var.project
  region  = var.region
}
