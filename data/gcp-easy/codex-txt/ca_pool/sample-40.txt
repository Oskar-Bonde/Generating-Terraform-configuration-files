terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block 
provider "google" {
  credentials = file("${path.module}/account.json")
  project     = "my-project"
  region      = "us-central1"
}

# create a CaPool resource
resource "google_compute_region_autoscaler" "default" {
  name = "my-autoscaler"
  region = "us-central1"
  target = "my-target"
  autoscaling_policy {
    max_replicas = 5
    min_replicas = 1
    cool_down_period_sec = 60
    cpu_utilization {
      target = 0.5
    }
  }
}
