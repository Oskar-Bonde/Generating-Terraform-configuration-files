terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

  credentials = file("account.json")
  project     = "my-project"
  region      = "us-central1"
}

  name = "default-firewall-policy"
  description = "Default firewall policy"
}
