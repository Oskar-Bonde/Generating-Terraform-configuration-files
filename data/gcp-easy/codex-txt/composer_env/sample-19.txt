terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block 
provider "google" {
  project = "my-project"
  region  = "us-central1"
}

resource "google_composer_environment" "my_environment" {
  name = "my-environment"
  config {
    node_count = 3
    node_config {
      machine_type = "n1-standard-1"
      disk_size_gb = 10
      location = "us-central1-a"
    }
  }
}
