terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block 
provider "google" {
  credentials = file("${path.module}/credentials.json")
  project     = "my-project"
  region      = "us-central1"
}

# create composer environment
resource "google_composer_environment" "my_composer_environment" {
  name = "my-composer-environment"
  config {
    node_count = 1
    node_config {
      machine_type = "n1-standard-1"
      disk_size_gb = 10
      network = "default"
    }
  }
}
