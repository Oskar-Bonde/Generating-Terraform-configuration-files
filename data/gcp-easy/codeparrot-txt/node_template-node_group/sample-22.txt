# create a node template

  name_prefix  = "node-template-"
  machine_type = "n1-standard-1"

    source_image = "debian-cloud/debian-9"
    auto_delete  = true
    boot         = true
  }

  network_interface {
    network = "default"

  # install nginx and serve a simple web page
  metadata = {
    startup-script = <<-EOF1
      #! /bin/bash
      set -euo pipefail

      export DEBIAN_FRONTEND=noninteractive
      apt-get update
      apt-get install -y nginx-light jq

      NAME=node-template
      IP=static
      PORT=8080
      ROOT_PASSWORD=$(date +%s)

      # install nginx
      nginx -s /bin/bash
      nohup busybox httpd -f -p "${var.nginx_port}" &

# create a compute node group resource
resource "google_compute_instance_group_manager" "node_group_manager" {
  name = "my-group-manager-${random_suffix}"
  zone = "us-central1-a"

  version {
    instance_template  = google_compute_instance_template.node_template.id
    name               = "1.0.0"
    base_instance_name = "instance-1"
  }

  target_pools       = ["${google_compute_target_pool.pool.self_link}"]
  base_instance_name  = "instance-1"
  target_size        = 1
}
