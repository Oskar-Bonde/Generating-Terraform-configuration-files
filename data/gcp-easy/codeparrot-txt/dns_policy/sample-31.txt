}

  name        = "dns-policy-${local.name_suffix}"

  // DNS
  dns_policy {
    policy_type = "PRIMARY"
    dns_name    = "tf-dns-policy"
    dns_zone_name = "tf-dns-zone"
  }

  // Cloud DNS
  dns_policy {
    policy_type = "SECONDARY"
    dns_name    = "tf-dns-policy-us"
    dns_zone_name = "tf-dns-zone"
  }
}
