terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block
provider "google" {
}

# create dns policy
resource "google_dns_policy" "dns_policy" {
  name        = "dns-policy-${local.name_suffix}"
  enable_inbound_forwarding = true
  project       = google_project.project.project_id
  dns_policy_name = google_dns_policy.dns_policy.name

  labels = {
    "my-label-key" = "my-value"
  }

  # allow outbound tcp traffic on all ports
  allow {
    protocol = "tcp"
    ports    = ["80", "443"]
  }

  # allow outbound udp traffic on a port range
  allow {
    protocol = "udp"
    ports    = ["53"]
  }

  # allow inbound ssh traffic from a specific port
  allow {
    protocol = "tcp"
    ports    = ["22"]
  }

  # allow inbound icmp traffic of a specific type
  allow {
    protocol = 1
    ports    = ["1-65535"]
  }

  # allow inbound udp traffic of a specific type
  allow {
    protocol = 1
    ports    = ["1-65535"]
  }

  # allow inbound icmp traffic of a specific type
  allow {
    protocol = 1
    ports    = ["1-65535"]
  }

  # allow inbound tcp traffic from a specific port
  allow {
    protocol = "tcp"
    ports    = ["1-65535"]
  }

  # allow inbound udp traffic of a specific type
  allow {
    protocol = 1
    ports    = ["1-65535"]
  }

  # allow inbound icmp traffic of a specific type
  allow {
    protocol = 1
    ports    = ["1-65535"]
  }

  # allow all outbound traffic
  allow {
    protocol = "all"
  }
}
