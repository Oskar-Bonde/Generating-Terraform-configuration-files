terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block
provider "google" {
}

# create dns policy
resource "google_dns_policy" "dns_policy" {
  name = "dns-policy-${random_suffix}"
  zone = "us-central1-a"

  # DNS policy
  rule {
    action = "deny"
    priority = 1
    match_rules {
      cidr_list = ["0.0.0.0/0"]
    }
  }

  # DNS zone
  rule {
    action = "deny"
    priority = 1
    match_rules {
      name = "*.test.com"
    }
  }
}
