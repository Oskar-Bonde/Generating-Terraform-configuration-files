terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block
provider "google" {
}

# create compute network resource 
resource "google_compute_network" "my_custom_network" {
  name = "my-custom-network-${random_suffix}"
  auto_create_subnetworks = "false"
  routing_mode = "REGIONAL"
  project = google_project.project.project_id

  depends_on = [google_project_service.compute]
}

# resource block: create a compute router
resource "google_compute_router" "router" {
  name = "my-router-${random_suffix}"
  network = google_compute_network.my_custom_network.name

  bgp {
    asn = 64514
  }
}
