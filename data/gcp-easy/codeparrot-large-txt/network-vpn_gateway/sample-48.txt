terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block
provider "google" {
}

# create compute network
resource "google_compute_network" "default" {
  name = "my-network-${random_suffix}"
  auto_create_subnetworks = "false"
}

# create a compute ha vpn gatewayvariable
resource "google_compute_ha_vpn_gateway" "ha_gateway" {
  name = "my-ha-vpn-gateway-${random_suffix}"
  network = google_compute_network.default.id
  region = "us-central1"
  vpn_gateway_interface = "eth-1"
  # Do not create a VPN gateway if the gateway is already created
  # with a different region
  vpn_gateway_ip_address = "192.168.0.1"
  # Set the static ip address for the VPN gateway
  static_ip_address_allocation = "dynamic"
  # Set the static ip address for the VPN gateway
  static_bats_ip_address = "192.168.2.1"
}
