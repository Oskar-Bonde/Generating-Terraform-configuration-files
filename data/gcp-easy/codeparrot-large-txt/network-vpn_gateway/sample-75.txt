terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block
provider "google" {
}

# create compute network
resource "google_compute_network" "default" {
  name = "my-network-${random_suffix}"
  auto_create_subnetworks = false
}

# create a compute ha vpn gatewayvariable
resource "google_compute_ha_vpn_gateway" "ha_gateway" {
  name = "my-ha-vpn-${random_suffix}"
  network = google_compute_network.default.id
  region = "us-central1"
  vpn_gateway_type = "HOP"
  # Do not create a VPN gateway if the gateway type is not HTTP
  # and VPN tunnel type is HTTP
  vpn_gateway = false
  # Do not create a VPN gateway if the gateway type is HTTP
  vpn_gateway_interface = "GKE_VPN_GW"
  # Do not create a VPN gateway if the gateway type is HTTP
  vpn_gateway_ip = "192.168.0.1"
}
