terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block
provider "google" {
}

# create compute network resource 
resource "google_compute_network" "my_custom_network" {
  name = "my-custom-network-${local.name_suffix}"
  auto_create_subnetworks = false
}

# resource block: create a compute routervariable
resource "google_compute_router_variable" "my_custom_router_variable" {
  name = "my-custom-router-variable-${local.name_suffix}"
  region = "us-central1"
  network = google_compute_network.my_custom_network.self_link

  ip_address = "192.168.0.1"
  type = "PRIMARY"
  router = google_compute_router.my_custom_router.name

  # Add a rule to allow HTTP(S) load balancing
  allow {
    protocol = "tcp"
    ports = ["80", "443"]
  }

  target_tags = ["my-custom-router-variable"]
}
