terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block
provider "google" {
}

# create compute network
resource "google_compute_network" "default" {
  name = "my-network-${local.name_suffix}"
}

# create compute firewallvariable
resource "google_compute_firewall_rule" "default" {
  name = "my-firewall-rule-${local.name_suffix}"
  network = "default"

  allow {
    protocol = "tcp"
    ports = ["80"]
  }

  target_tags = ["web"]
}
