terraform {
  required_providers {
    google = {
      source = "hashicorp/google"
    }
  }
}

# google provider block
provider "google" {
}

# create dns policy
resource "google_dns_policy" "dns_policy" {
  name        = "dns-policy-${random_suffix}"
  zone        = "us-central1-a"
  force_destroy = true

  rules {
    # allow outbound tcp traffic on all ports
    allow_outbound_traffic = true
  }

  networks {
    network_url = google_compute_network.my-custom-network.id
  }
}
