# raw-data/hashicorp-certified-terraform\03-Terraform-Fundamental-Blocks\03-04-Providers-Dependency-Lock-File\terraform-manifests
# Terraform block with AWS and random provider
terraform {
  required_version = "~> 0.14.6"
  required_providers {
    aws = {
      source  = "hashicorp/aws"
      version = ">= 2.0.0"
    }
    random = {
      source  = "hashicorp/random"
      version = "3.0.0"
    }
  }
}

# AWS provider block in region us-east-1  and profile set as default
provider "aws" {
  region = "us-east-1"
  profile = "default"
}

# Resource block to create a random pet name of length 5 with separator -
resource "random_pet" "petname" {
  length    = 5
  separator = "-"
}

# Resource Block: Create AWS S3 Bucket with bucket set as random pet name
resource "aws_s3_bucket" "sample" {
  bucket = random_pet.petname.id
  acl    = "public-read"
  region = "us-east-1" 
}
