# raw-data/hashicorp-certified-terraform\03-Terraform-Fundamental-Blocks\03-01-Terraform-Block
# Terraform block with AWS and random provider as well as a backend S3 bucket in us-east-1, bucket called mybucket and key in path "key"
terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
    }
    random = {
      source = "hashicorp/random"
    }
  }
  backend "s3" {
    bucket = "mybucket"
    key    = "key"
    region = "us-east-1"

  }
}

# Provider block with AWS in us-east-1
provider "aws" {
  profile = "default" 
  region  = "us-east-1"
}
